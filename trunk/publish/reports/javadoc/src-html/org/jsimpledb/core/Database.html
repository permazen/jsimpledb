039/**
040 * Provides an object database abstraction on top of a key/value database.
041 *
042 * <p>
043 * Includes support for:
044 * <ul>
045 *  <li>Objects and fields defined by a {@link SchemaModel}, with positive schema verification</li>
046 *  <li>Simple values fields containing any atomic type, reference or custom {@link FieldType}</li>
047 *  <li>Complex fields of type {@link java.util.List}, {@link java.util.NavigableSet}, and {@link java.util.NavigableMap}</li>
048 *  <li>Invertable reference fields with strong referential integrity and configurable delete cascading</li>
049 *  <li>Configurable indexing of any simple field or complex sub-field</li>
050 *  <li>Composite indexes on multiple simple fields</li>
051 *  <li>Notification of object creation and deletion</li>
052 *  <li>Notification of object field changes, as seen through an arbitrary path of references</li>
053 *  <li>Automatic schema tracking and object versioning with schema change notification support</li>
054 * </ul>
055 *
056 * <p>
057 * See {@link Transaction} for further details on the above functionality.
058 * </p>
059 *
060 * <p>
061 * This class defines an abstraction layer that usually sits below a {@link org.jsimpledb.JSimpleDB} but is completely
062 * independent of {@link org.jsimpledb.JSimpleDB} and can be used on its own.
063 * Compared to {@link org.jsimpledb.JSimpleDB}, a {@link Database} has these differences:
064 * <ul>
065 *  <li>A {@link SchemaModel} must be explicitly provided to define the schema in use, whereas when using a
066 *      {@link org.jsimpledb.JSimpleDB} the schema is derived automatically from annotated Java model classes.</li>
067 *  <li>Object references are represented by {@link ObjId}s instead of Java objects, and there is no notion of object sub-type.
068 *      However, reference fields may be configured with a restricted set of referrable types.</li>
069 *  <li>All object types and fields are referenced by explicit storage ID.</li>
070 *  <li>Enum values are represented by {@link EnumValue} objects.</li>
071 *  <li>There is no automatic validation support.</li>
072 * </ul>
073 *
074 * @see Transaction
075 * @see org.jsimpledb
076 */
077public class Database {

184    /**
185     * Create a new {@link Transaction} on this database and use the specified schema version to access objects and fields.
186     *
187     * <p>
188     * <b>Schema Versions</b>
189     * </p>
190     *
191     * <p>
192     * Within each {@link Database} is stored a record of all schema versions previously used with the database.
193     * When creating a new transaction, the caller provides an expected schema version and corresponding {@link SchemaModel}.
194     * Both of these are optional: a schema version of zero means "use the highest version recorded in the
195     * database", and a null {@link SchemaModel} means "use the {@link SchemaModel} already recorded in the database under
196     * {@code version}".
197     * </p>
198     *
199     * <p>
200     * When this method is invoked, the following checks are applied:
201     * <ul>
202     *  <li>If a schema with version number {@code version != 0} is recorded in the database, and {@code schemaModel} is null or
203     *      matches it, then this method succeeds, and the {@link Transaction} will use that schema.</li>
204     *  <li>If a schema with version number {@code version} (or the highest numbered schema if {@code version == 0})
205     *      is recorded in the database, and {@code schemaModel} is not null and does not match it, then this method fails
206     *      and throws a {@link SchemaMismatchException}.</li>
207     *  <li>If {@code allowNewSchema} is false, and no schema with version number {@code version != 0} has yet been
208     *      recorded in the database, then this method fails and throws a {@link SchemaMismatchException}.</li>
209     *  <li>If {@code allowNewSchema} is true, and no schema with version number {@code version != 0} has yet been
210     *      recorded in the database, then if {@code schemaModel} is null a {@link SchemaMismatchException} is thrown;
211     *      otherwise {@code schemaModel} is checked for compabitility with the schemas previously recorded in the database;
212     *      if compatible, this method succeeds, {@code schema} is recorded in the database under the new version number
213     *      {@code version}, and the {@link Transaction} will use schema version {@code version};
214     *      otherwise a {@link SchemaMismatchException} is thrown.</li>
215     *  <li>If the database is uninitialized and {@code version == 0} or {@code schemaModel} is null,
216     *      a {@link SchemaMismatchException} is thrown.</li>
217     * </ul>
218     *
219     * <p>
220     * For two schemas to "match", they must be identical in all respects, except that object, field, and index names may differ.
221     * In the core API, objects and fields are identified by storage ID, not name.
222     * </p>
223     *
224     * <p>
225     * Schemas must also be compatible with all other schemas previously recorded in the database.
226     * Basically this means storage IDs must be used consistently from a structural point of view:
227     * <ul>
228     *  <li>Once a storage ID is assigned, it cannot be re-assigned to a different type of item (object or field).</li>
229     *  <li>Fields must have a consistent type and structural parent (object type or complex field).</li>
230     * </ul>
231     *
232     * <p>
233     * However, object types and fields may be added or removed across schema versions, field indexing may change,
234     * and reference field {@link DeleteAction}s may change.
235     * </p>
236     *
237     * <p>
238     * <b>Object Versions</b>
239     * </p>
240     *
241     * <p>
242     * Each object in a {@link Database} contains an internal version number that indicates its current schema version;
243     * this in turn dictates what fields that object contains.
244     * </p>
245     *
246     * <p>
247     * When an object is accessed during a {@link Transaction}, the object's version is compared to the {@code version} associated
248     * with that {@link Transaction}. If the versions are the same, no version change occurs and fields are accessed normally.
249     * </p>
250     *
251     * <p>
252     * If the object has a version {@code oldVersion} different from {@code version}, then depending on which {@link Transaction}
253     * method is invoked, the object version may be automatically updated to {@code version}. This will cause fields to be added
254     * or removed, as follows:
255     * <ul>
256     *  <li>Fields that are common to both schema versions remain unchanged (necessarily such fields have the same storage ID,
257     *      type, and structural parent).</li>
258     *  <li>Fields that exist in {@code oldVersion} but not in {@code version} are removed.</li>
259     *  <li>Fields that exist in {@code version} but not in {@code oldVersion} are initialized to their default values.</li>
260     *  <li>All {@link VersionChangeListener}s registered with the {@link Transaction} are notified.</li>
261     * </ul>
262     *
263     * <p>
264     * Note that compatibility between schema versions does not depend on the field name, nor does it depend on whether the field
265     * is indexed, or its {@link DeleteAction} (for reference fields). A field's index may be added or removed between schema
266     * versions without losing information, however, querying a field's index will only return those objects whose schema
267     * version corresponds to a schema in which the field is indexed. Similarly, the {@link DeleteAction} taken when a
268     * referenced object is deleted depends on the {@link DeleteAction} configured in the schema version of the object
269     * containing the reference.
270     * </p>
271     *
272     * <p>
273     * Note that an object's current schema version can go up as well as down, may change non-consecutively, and in fact
274     * nothing requires schema version numbers to be consecutive.
275     * </p>
276     *
277     * @param schemaModel schema to use with the new transaction, or null to use the schema already recorded in the database
278     * @param version the schema version number corresponding to {@code schemaModel}, or zero to use the highest recorded version
279     * @param allowNewSchema whether creating a new schema version is allowed
280     * @return newly created transaction
281     * @throws IllegalArgumentException if {@code version} is less than zero
282     * @throws InvalidSchemaException if {@code schemaModel} is invalid (i.e., does not pass validation checks)
283     * @throws SchemaMismatchException if {@code schemaModel} does not match schema version {@code version}
284     *  as recorded in the database
285     * @throws SchemaMismatchException if schema version {@code version} is not recorded in the database
286     *  and {@code allowNewSchema} is false
287     * @throws SchemaMismatchException if schema version {@code version} is not recorded in the database,
288     *  {@code allowNewSchema} is true, but {@code schemaModel} is incompatible with one or more other schemas
289     *  already recorded in the database (i.e., the same storage ID is used inconsistently between schema versions)
290     * @throws SchemaMismatchException
291     *  if the database is uninitialized and {@code version == 0} or {@code schemaModel} is null
292     * @throws InconsistentDatabaseException if inconsistent or invalid schema information is detected in the database
293     * @throws InconsistentDatabaseException if an uninitialized database is encountered but the database is not empty
294     * @throws IllegalStateException if no underlying {@link KVDatabase} has been configured for this instance
295     */
296    public Transaction createTransaction(final SchemaModel schemaModel, int version, final boolean allowNewSchema) {
297
298        // Sanity check
299        if (version < 0)
300            throw new IllegalArgumentException("invalid schema version: " + version);
301
302        // Validate schema
303        if (schemaModel != null)
304            schemaModel.validate();
305
306        // Open KV transaction
307        final KVTransaction kvt = this.kvdb.createTransaction();
308        boolean success = false;
309        if (this.log.isTraceEnabled()) {
310            this.log.trace("creating transaction using "
311              + (version != 0 ? "schema version " + version : "highest recorded schema version"));
312        }
313        try {
314
315            // Get iterator over meta-data key/value pairs
316            final byte[] metaDataPrefix = new byte[] { METADATA_PREFIX };
317            final Iterator<KVPair> metaDataIterator = kvt.getRange(metaDataPrefix,
318              ByteUtil.getKeyAfterPrefix(metaDataPrefix), false);
319
320            // Get format version; it should be first; if not found, database is uninitialized (and should be empty)
321            byte[] formatVersionBytes = null;
322            if (metaDataIterator.hasNext()) {
323                final KVPair pair = metaDataIterator.next();
324                if (!Arrays.equals(pair.getKey(), FORMAT_VERSION_KEY)) {
325                    throw new InconsistentDatabaseException("database is uninitialized but contains unrecognized garbage (key "
326                      + ByteUtil.toString(pair.getKey()) + ")");
327                }
328                formatVersionBytes = pair.getValue();
329            }
330
331            // Get database format object; check for an uninitialized database
332            final boolean uninitialized = formatVersionBytes == null;
333            final int formatVersion;
334            if (uninitialized) {
335
336                // Sanity checks
337                if (kvt.getAtLeast(new byte[0]) != null)
338                    throw new InconsistentDatabaseException("database is uninitialized but contains unrecognized garbage");
339                if (kvt.getAtMost(new byte[] { (byte)0xff }) != null)
340                    throw new InconsistentDatabaseException("inconsistent results from getAtLeast() and getAtMost()");
341                if (kvt.getRange(new byte[0], new byte[] { (byte)0xff }, false).hasNext())
342                    throw new InconsistentDatabaseException("inconsistent results from getAtLeast() and getRange()");
343                this.checkAddNewSchema(schemaModel, version, allowNewSchema);
344
345                // Initialize database
346                formatVersion = CURRENT_FORMAT_VERSION;
347                this.log.info("detected an uninitialized database; initializing with format version " + formatVersion);
348                final ByteWriter writer = new ByteWriter();
349                UnsignedIntEncoder.write(writer, CURRENT_FORMAT_VERSION);
350                kvt.put(FORMAT_VERSION_KEY.clone(), writer.getBytes());
351
352                // Sanity check again
353                formatVersionBytes = kvt.get(FORMAT_VERSION_KEY.clone());
354                if (formatVersionBytes == null || ByteUtil.compare(formatVersionBytes, writer.getBytes()) != 0)
355                    throw new InconsistentDatabaseException("database failed basic read/write test");
356                final KVPair lower = kvt.getAtLeast(new byte[0]);
357                if (lower == null || !lower.equals(new KVPair(FORMAT_VERSION_KEY, writer.getBytes())))
358                    throw new InconsistentDatabaseException("database failed basic read/write test");
359                final KVPair upper = kvt.getAtMost(new byte[] { (byte)0xff });
360                if (upper == null || !upper.equals(new KVPair(FORMAT_VERSION_KEY, writer.getBytes())))
361                    throw new InconsistentDatabaseException("database failed basic read/write test");
362            } else {
363                try {
364                    formatVersion = UnsignedIntEncoder.decode(formatVersionBytes);
365                } catch (IllegalArgumentException e) {
366                    throw new InconsistentDatabaseException("database contains invalid encoded format version "
367                      + ByteUtil.toString(formatVersionBytes) + " under key " + ByteUtil.toString(FORMAT_VERSION_KEY));
368                }
369            }
370            final boolean compressedSchemaXML;
371            switch (formatVersion) {
372            case FORMAT_VERSION_1:
373            case FORMAT_VERSION_2:
374                compressedSchemaXML = formatVersion >= FORMAT_VERSION_2;
375                break;
376            default:
377                throw new InconsistentDatabaseException("database contains unrecognized format version "
378                  + formatVersion + " under key " + ByteUtil.toString(FORMAT_VERSION_KEY));
379            }
380
381            // There should not be any other meta data prior to recorded schemas
382            if (metaDataIterator.hasNext()) {
383                final KVPair pair = metaDataIterator.next();
384                if (ByteUtil.compare(pair.getKey(), SCHEMA_KEY_PREFIX) < 0) {
385                    throw new InconsistentDatabaseException("database contains unrecognized garbage at key "
386                      + ByteUtil.toString(pair.getKey()));
387                }
388            }
389
390            // Check schema
391            Schemas schemas = null;
392            boolean firstAttempt = true;
393            while (true) {
394
395                // Get iterator over schema key/value pairs
396                final Iterator<KVPair> schemaIterator = kvt.getRange(SCHEMA_KEY_PREFIX.clone(),
397                  ByteUtil.getKeyAfterPrefix(SCHEMA_KEY_PREFIX), false);
398
399                // Read recorded database schema versions - should immediately follow FORMAT_VERSION_KEY
400                final TreeMap<Integer, byte[]> bytesMap = new TreeMap<>();
401                while (schemaIterator.hasNext()) {
402                    final KVPair pair = schemaIterator.next();
403
404                    // Sanity check
405                    if (ByteUtil.compare(pair.getKey(), SCHEMA_KEY_PREFIX) < 0) {
406                        throw new InconsistentDatabaseException("database contains unrecognized garbage key "
407                          + ByteUtil.toString(pair.getKey()));
408                    }
409
410                    // Stop at end of recorded schemas
411                    if (!ByteUtil.isPrefixOf(SCHEMA_KEY_PREFIX, pair.getKey()))
412                        break;
413
414                    // Decode schema version and get XML
415                    final int vers = UnsignedIntEncoder.read(new ByteReader(pair.getKey(), SCHEMA_KEY_PREFIX.length));
416                    if (vers == 0)
417                        throw new InconsistentDatabaseException("database contains an invalid schema version zero");
418                    bytesMap.put(vers, pair.getValue());
419                }
420
421                // Read and decode database schemas, avoiding rebuild if possible
422                schemas = this.lastSchemas;
423                if (schemas != null && !schemas.isSameVersions(bytesMap))
424                    schemas = null;
425                if (schemas == null) {
426                    try {
427                        schemas = this.buildSchemas(bytesMap, compressedSchemaXML);
428                    } catch (IllegalArgumentException e) {
429                        if (firstAttempt)
430                            throw new InconsistentDatabaseException("database contains invalid schema information", e);
431                        else
432                            throw new InvalidSchemaException("schema is not valid: " + e.getMessage(), e);
433                    }
434                }
435
436                // If no version specified, assume the highest recorded version
437                if (version == 0 && !bytesMap.isEmpty())
438                    version = bytesMap.lastKey();
439
440                // If transaction schema was not found in the database, add it and retry
441                if (!bytesMap.containsKey(version)) {
442
443                    // Log it
444                    if (bytesMap.isEmpty()) {
445                        if (!uninitialized)
446                            throw new InconsistentDatabaseException("database is initialized but contains zero schema versions");
447                    } else {
448                        this.log.info("schema version " + version
449                          + " not found in database; known versions are " + bytesMap.keySet());
450                    }
451
452                    // Check whether we can add a new schema version
453                    this.checkAddNewSchema(schemaModel, version, allowNewSchema);
454
455                    // Record new schema in database
456                    this.log.info("recording new schema version " + version + " into database");
457                    this.writeSchema(kvt, version, schemaModel, compressedSchemaXML);
458
459                    // Try again
460                    schemas = null;
461                    firstAttempt = false;
462                    continue;
463                }
464
465                // Compare transaction schema with the schema of the same version found in the database
466                if (this.log.isTraceEnabled())
467                    this.log.trace("found schema version " + version + " in database; known versions are " + bytesMap.keySet());
468                final SchemaModel dbSchemaModel = schemas.getVersion(version).getSchemaModel();
469                if (schemaModel != null) {
470                    if (!schemaModel.isCompatibleWith(dbSchemaModel)) {
471                        final Diffs diffs = schemaModel.differencesFrom(dbSchemaModel);
472                        this.log.error("schema mismatch:\n=== Database schema ===\n{}\n=== Provided schema ===\n{}"
473                          + "\n=== Differences ===\n{}", dbSchemaModel, schemaModel, diffs);
474                        throw new IllegalArgumentException("the provided transaction schema does not match the schema with version "
475                          + version + " that is already recorded in the database:\n" + diffs);
476                    } else if (this.log.isTraceEnabled() && !schemaModel.equals(dbSchemaModel)) {
477                        final Diffs diffs = schemaModel.differencesFrom(dbSchemaModel);
478                        this.log.trace("the provided schema differs from, but is compatible with, the database schema:\n{}", diffs);
479                    }
480                }
481                break;
482            }
483
484            // Save schema for next time
485            this.lastSchemas = schemas;
486
487            // Create transaction
488            final Transaction tx = new Transaction(this, kvt, schemas, version);
489            success = true;
490            return tx;
491        } finally {
492            if (!success) {
493                try {
494                    kvt.rollback();
495                } catch (KVTransactionException e) {
496                    // ignore
497                }
498            }
499        }
500    }

502    /**
503     * Validate a {@link SchemaModel}.
504     *
505     * <p>
506     * This method only performs "static" checks; it does not access the database and therefore
507     * does not validate the schema against existing schema versions previously recorded.
508     * It does however use the {@link FieldTypeRegistry} associated with this instance to look up field types.
509     * </p>
510     *
511     * <p>
512     * To validate a schema against the database contents as well, simply attempt to create a transaction
513     * via {@link #createTransaction createTransaction()}.
514     * </p>
515     *
516     * @param schemaModel schema to validate
517     * @throws InvalidSchemaException if {@code schemaModel} is invalid
518     * @throws IllegalArgumentException if {@code schemaModel} is null
519     */
520    public void validateSchema(SchemaModel schemaModel) {
521
522        // Sanity check
523        if (schemaModel == null)
524            throw new IllegalArgumentException("null schemaModel");
525
526        // Validate
527        schemaModel.validate();
528        try {
529            new Schema(1, new byte[0], schemaModel, this.fieldTypeRegistry);
530        } catch (IllegalArgumentException e) {
531            throw new InvalidSchemaException("invalid schema: " + e.getMessage(), e);
532        }
533    }
534
535    private void checkAddNewSchema(SchemaModel schemaModel, int version, boolean allowNewSchema) {
536        if (version == 0)
537            throw new SchemaMismatchException("database is uninitialized and no schema version was provided");
538        if (schemaModel == null) {
539            throw new SchemaMismatchException("schema version " + version
540              + " was not found in database, and no schema model was provided");
541        }
542        if (!allowNewSchema) {
543            throw new SchemaMismatchException("schema version " + version
544              + " was not found in database, and recording a new schema version is disabled in this transaction");
545        }
546    }

548    void copyMetaData(Transaction src, KVStore dst) {
549        for (Iterator<KVPair> i = src.kvt.getRange(new byte[] { METADATA_PREFIX }, VERSION_INDEX_PREFIX.clone(), false);
550          i.hasNext(); ) {
551            final KVPair pair = i.next();
552            dst.put(pair.getKey(), pair.getValue());
553        }
554    }

556    void reset(SnapshotTransaction tx) {
557        tx.kvt.removeRange(VERSION_INDEX_PREFIX.clone(), null);
558    }

560    static byte[] buildVersionIndexKey(ObjId id, int version) {
561        final ByteWriter writer = new ByteWriter(VERSION_INDEX_PREFIX.length + 1 + ObjId.NUM_BYTES);
562        writer.write(VERSION_INDEX_PREFIX);
563        UnsignedIntEncoder.write(writer, version);
564        id.writeTo(writer);
565        return writer.getBytes();
566    }

573    /**
574     * Build {@link Schemas} object from a schema version XMLs.
575     *
576     * @throws InconsistentDatabaseException if any recorded schema version is invalid
577     */
578    private Schemas buildSchemas(SortedMap<Integer, byte[]> bytesMap, boolean compressed) {
579        final TreeMap<Integer, Schema> versionMap = new TreeMap<>();
580        for (Map.Entry<Integer, byte[]> entry : bytesMap.entrySet()) {
581            final int version = entry.getKey();
582            final byte[] bytes = entry.getValue();
583            final SchemaModel schemaModel;
584            try {
585                schemaModel = this.decodeSchema(bytes, compressed);
586            } catch (InvalidSchemaException e) {
587                throw new InconsistentDatabaseException("found invalid schema version " + version + " recorded in database", e);
588            }
589            if (this.log.isTraceEnabled())
590                this.log.trace("read schema version {} from database:\n{}", version, schemaModel);
591            versionMap.put(version, new Schema(version, bytes, schemaModel, this.fieldTypeRegistry));
592        }
593        return new Schemas(versionMap);
594    }
595
596    /**
597     * Decode and validate schema XML.
598     *
599     * @throws InvalidSchemaException if schema is invalid
600     */
601    private SchemaModel decodeSchema(byte[] value, boolean compressed) {
602
603        // Decompress
604        if (compressed) {
605            try {
606                final Inflater inflater = new Inflater(true);
607                inflater.setDictionary(SCHEMA_XML_COMPRESSION_DICTIONARY);
608                inflater.setInput(Bytes.concat(value, new byte[1]));
609                final ByteArrayOutputStream decompressed = new ByteArrayOutputStream();
610                final byte[] temp = new byte[100];
611                int r;
612                while ((r = inflater.inflate(temp)) != 0)
613                    decompressed.write(temp, 0, r);
614                if (!inflater.finished())
615                    throw new RuntimeException("internal error: inflater did not finish");
616                inflater.end();
617                value = decompressed.toByteArray();
618            } catch (DataFormatException e) {
619                throw new InvalidSchemaException("error in compressed data", e);
620            }
621        }
622
623        // Decode XML
624        try {
625            return SchemaModel.fromXML(new ByteArrayInputStream(value));
626        } catch (IOException e) {
627            throw new RuntimeException("unexpected exception", e);
628        }
629    }
630
631    /**
632     * Record the given schema into the database.
633     */
634    private void writeSchema(KVTransaction kvt, int version, SchemaModel schemaModel, boolean compress) {
635
636        // Encode as XML
637        final ByteArrayOutputStream buf = new ByteArrayOutputStream();
638        try {
639            schemaModel.toXML(buf, false);
640        } catch (IOException e) {
641            throw new RuntimeException("unexpected exception", e);
642        }
643        byte[] value = buf.toByteArray();
644
645        // Compress
646        if (compress) {
647            final Deflater deflater = new Deflater(Deflater.BEST_COMPRESSION, true);
648            deflater.setDictionary(SCHEMA_XML_COMPRESSION_DICTIONARY);
649            deflater.setInput(value);
650            deflater.finish();
651            final ByteArrayOutputStream compressed = new ByteArrayOutputStream();
652            final byte[] temp = new byte[100];
653            int r;
654            while ((r = deflater.deflate(temp)) != 0)
655                compressed.write(temp, 0, r);
656            if (!deflater.finished())
657                throw new RuntimeException("internal error: deflater did not finish");
658            deflater.end();
659            value = compressed.toByteArray();
660        }
661
662        // Write schema
663        kvt.put(this.getSchemaKey(version), value);
664    }
665
666    /**
667     * Delete a schema version. Caller must verify no objects exist.
668     */
669    void deleteSchema(KVTransaction kvt, int version) {
670        kvt.remove(this.getSchemaKey(version));
671    }
672
673    private byte[] getSchemaKey(int version) {
674        final ByteWriter writer = new ByteWriter();
675        writer.write(SCHEMA_KEY_PREFIX);
676        UnsignedIntEncoder.write(writer, version);
677        return writer.getBytes();
678    }
