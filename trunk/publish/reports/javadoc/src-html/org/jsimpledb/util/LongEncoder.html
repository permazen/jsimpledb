008package org.jsimpledb.util;
009
010import java.io.EOFException;
011import java.io.IOException;
012import java.io.InputStream;
013import java.io.OutputStream;
014import java.nio.ByteBuffer;
015import java.util.Arrays;
016
017/**
018 * Encodes {@code long} values to/from binary, preserving sort order, and such that the length of the
019 * encoding is optimized for values near zero.
020 *
021 * <p>
022 * Encoded values are guaranteed to not start with {@code 0x00} or {@code 0xff}.
023 * </p>
024 */
025public final class LongEncoder {
026
027    /**
028     * Maximum possible length of an encoded value.
029     */
030    public static final int MAX_ENCODED_LENGTH = 9;
031
032    /**
033     * Minimum value for the first byte of a single byte encoded value.
034     * Lower values indicate a multiple byte encoded negative value.
035     */
036    public static final int MIN_SINGLE_BYTE_ENCODED = 0x09;        // values 0x01 ... 0x08 prefix negative values
037
038    /**
039     * Maximum value for the first byte of a single byte encoded value.
040     * Higher values indicate a multiple byte encoded positive value.
041     */
042    public static final int MAX_SINGLE_BYTE_ENCODED = 0xf6;        // values 0xf7 ... 0xfe prefix positive values
043
044    /**
045     * Adjustment applied to single byte encoded values before encoding.
046     */
047    public static final int ZERO_ADJUST = 127;                     // single byte value that represents zero
048
049    /**
050     * Minimum value that can be encoded as a single byte.
051     */
052    public static final int MIN_SINGLE_BYTE_VALUE = MIN_SINGLE_BYTE_ENCODED - ZERO_ADJUST;          // -118
053
054    /**
055     * Maximum value that can be encoded as a single byte.
056     */
057    public static final int MAX_SINGLE_BYTE_VALUE = MAX_SINGLE_BYTE_ENCODED - ZERO_ADJUST;          // 119
058
059    /**
060     * Adjustment applied to multi-byte encoded negative values before encoding.
061     */
062    public static final int NEGATIVE_ADJUST = -MIN_SINGLE_BYTE_VALUE;                               // 118
063
064    /**
065     * Adjustment applied to multi-byte encoded positive values before encoding.
066     */
067    public static final int POSITIVE_ADJUST = -(MAX_SINGLE_BYTE_VALUE + 1);                         // -120
068
069    // Cutoff values at which the encoded length changes (this field is package private for testing purposes)
070    static final long[] CUTOFF_VALUES = new long[] {
071        0xff00000000000000L - NEGATIVE_ADJUST,      // [ 0] requires 8 bytes
072        0xffff000000000000L - NEGATIVE_ADJUST,      // [ 1] requires 7 bytes
073        0xffffff0000000000L - NEGATIVE_ADJUST,      // [ 2] requires 6 bytes
074        0xffffffff00000000L - NEGATIVE_ADJUST,      // [ 3] requires 5 bytes
075        0xffffffffff000000L - NEGATIVE_ADJUST,      // [ 4] requires 4 bytes
076        0xffffffffffff0000L - NEGATIVE_ADJUST,      // [ 5] requires 3 bytes
077        0xffffffffffffff00L - NEGATIVE_ADJUST,      // [ 6] requires 2 bytes
078        MIN_SINGLE_BYTE_VALUE,                      // [ 7] requires 1 byte
079        MAX_SINGLE_BYTE_VALUE + 1,                  // [ 8] requires 2 bytes
080        0x0000000000000100L - POSITIVE_ADJUST,      // [ 9] requires 3 bytes
081        0x0000000000010000L - POSITIVE_ADJUST,      // [10] requires 4 bytes
082        0x0000000001000000L - POSITIVE_ADJUST,      // [11] requires 5 bytes
083        0x0000000100000000L - POSITIVE_ADJUST,      // [12] requires 6 bytes
084        0x0000010000000000L - POSITIVE_ADJUST,      // [13] requires 7 bytes
085        0x0001000000000000L - POSITIVE_ADJUST,      // [14] requires 8 bytes
086        0x0100000000000000L - POSITIVE_ADJUST,      // [15] requires 9 bytes
087    };
088
089    private LongEncoder() {
090    }
091
092    /**
093     * Encode the given value.
094     *
095     * @param value value to encode
096     * @return encoded bytes
097     */
098    public static byte[] encode(long value) {
099        final ByteWriter writer = new ByteWriter(LongEncoder.encodeLength(value));
100        LongEncoder.write(writer, value);
101        return writer.getBytes();
102    }
103
104    /**
105     * Decode the given value.
106     *
107     * @param data encoded value
108     * @return decoded value
109     * @throws IllegalArgumentException if {@code bytes} contains an invalid encoding, or extra trailing garbage
110     */
111    public static long decode(byte[] data) {
112        final ByteReader reader = new ByteReader(data);
113        final long value = LongEncoder.read(reader);
114        if (reader.remain() > 0)
115            throw new IllegalArgumentException("encoded value contains extra trailing garbage");
116        return value;
117    }
118
119    /**
120     * Encode the given value to the output.
121     *
122     * @param writer destination for the encoded value
123     * @param value value to encode
124     * @throws NullPointerException if {@code writer} is null
125     */
126    public static void write(ByteWriter writer, long value) {
127        writer.makeRoom(MAX_ENCODED_LENGTH);
128        writer.len += LongEncoder.encode(value, writer.buf, writer.len);
129    }
130
131    /**
132     * Encode the given value and write it to the given {@link OutputStream}.
133     *
134     * @param out destination for the encoded value
135     * @param value value to encode
136     * @throws IOException if an I/O error occurs
137     * @throws NullPointerException if {@code out} is null
138     */
139    public static void write(OutputStream out, long value) throws IOException {
140        final byte[] array = new byte[LongEncoder.MAX_ENCODED_LENGTH];
141        final int nbytes = LongEncoder.encode(value, array, 0);
142        out.write(array, 0, nbytes);
143    }
144
145    /**
146     * Encode the given value and write it to the given {@link ByteBuffer}.
147     *
148     * @param buf destination for the encoded value
149     * @param value value to encode
150     * @throws java.nio.BufferOverflowException if {@code buf} overflows
151     * @throws java.nio.ReadOnlyBufferException if {@code buf} is read-only
152     * @throws NullPointerException if {@code buf} is null
153     */
154    public static void write(ByteBuffer buf, long value) {
155        final byte[] array = new byte[LongEncoder.MAX_ENCODED_LENGTH];
156        final int nbytes = LongEncoder.encode(value, array, 0);
157        buf.put(array, 0, nbytes);
158    }
159
160    /**
161     * Read and decode a value from the input.
162     *
163     * @param reader input holding an encoded value
164     * @return the decoded value
165     * @throws IllegalArgumentException if an invalid encoding is encountered
166     * @throws IllegalArgumentException if the encoded value is truncated
167     * @throws NullPointerException if {@code reader} is null
168     */
169    public static long read(ByteReader reader) {
170        try {
171            int first = reader.readByte();
172            if (first < MIN_SINGLE_BYTE_ENCODED) {
173                if (first == 0x00)
174                    throw new IllegalArgumentException("invalid encoded value starting with 0x00");
175                long value = ~0L;
176                while (first++ < MIN_SINGLE_BYTE_ENCODED)
177                    value = (value << 8) | reader.readByte();
178                return value - NEGATIVE_ADJUST;
179            }
180            if (first > MAX_SINGLE_BYTE_ENCODED) {
181                if (first == 0xff)
182                    throw new IllegalArgumentException("invalid encoded value starting with 0xff");
183                long value = 0L;
184                while (first-- > MAX_SINGLE_BYTE_ENCODED)
185                    value = (value << 8) | reader.readByte();
186                return value - POSITIVE_ADJUST;
187            }
188            return (byte)(first - ZERO_ADJUST);
189        } catch (IndexOutOfBoundsException e) {
190            throw new IllegalArgumentException("encoded value is truncated", e);
191        }
192    }
193
194    /**
195     * Read and decode a value from the given {@link InputStream}.
196     *
197     * @param input input source for the encoded value
198     * @return the decoded value
199     * @throws IOException if an I/O error occurs
200     * @throws EOFException if an unexpected EOF is encountered
201     * @throws IllegalArgumentException if an invalid encoding is encountered
202     * @throws NullPointerException if {@code input} is null
203     */
204    public static long read(InputStream input) throws IOException {
205        final int first = input.read();
206        if (first == -1)
207            throw new EOFException();
208        final byte[] array = new byte[LongEncoder.decodeLength(first)];
209        array[0] = (byte)first;
210        for (int i = 1; i < array.length; i++) {
211            final int next = input.read();
212            if (next == -1)
213                throw new EOFException();
214            array[i] = (byte)next;
215        }
216        return LongEncoder.read(new ByteReader(array));
217    }
218
219    /**
220     * Read and decode a value from the given {@link ByteBuffer}.
221     *
222     * @param buf input source for the encoded value
223     * @return the decoded value
224     * @throws java.nio.BufferUnderflowException if {@code buf} underflows
225     * @throws IllegalArgumentException if an invalid encoding is encountered
226     * @throws NullPointerException if {@code buf} is null
227     */
228    public static long read(ByteBuffer buf) {
229        final byte first = buf.get();
230        final byte[] array = new byte[LongEncoder.decodeLength(first)];
231        array[0] = first;
232        if (array.length > 1)
233            buf.get(array, 1, array.length - 1);
234        return LongEncoder.read(new ByteReader(array));
235    }
236
237    /**
238     * Skip a value from the input.
239     *
240     * @param reader input holding an encoded value
241     * @throws IllegalArgumentException if the first byte is {@code 0xff}
242     */
243    public static void skip(ByteReader reader) {
244        final int first = reader.readByte();
245        if (first == 0x00 || first == 0xff)
246            throw new IllegalArgumentException("invalid encoded value starting with 0x" + Integer.toHexString(first));
247        reader.skip(LongEncoder.decodeLength(first) - 1);
248    }
249
250    /**
251     * Determine the length (in bytes) of an encoded value based on the first byte.
252     *
253     * @param first first byte of encoded value (in lower eight bits; other bits are ignored)
254     * @return the length of the encoded value (including {@code first})
255     * @throws IllegalArgumentException if the first byte is {@code 0xff}
256     */
257    public static int decodeLength(int first) {
258        first &= 0xff;
259        if (first == 0x00 || first == 0xff)
260            throw new IllegalArgumentException("invalid encoded value starting with 0x" + Integer.toHexString(first));
261        if (first < MIN_SINGLE_BYTE_ENCODED)
262            return 1 + MIN_SINGLE_BYTE_ENCODED - first;
263        if (first > MAX_SINGLE_BYTE_ENCODED)
264            return 1 + first - MAX_SINGLE_BYTE_ENCODED;
265        return 1;
266    }
267
268    /**
269     * Determine the length (in bytes) of the encoded value.
270     *
271     * @param value value to encode
272     * @return the length of the encoded value, a value between one and {@link #MAX_ENCODED_LENGTH}
273     */
274    public static int encodeLength(long value) {
275        int index = Arrays.binarySearch(CUTOFF_VALUES, value);
276        if (index < 0)
277            index = ~index - 1;
278        return index < 8 ? 8 - index : index - 6;
279    }
280
281    /**
282     * Encode the given value and write the encoded bytes into the given buffer.
283     *
284     * @param value value to encode
285     * @param buf output buffer
286     * @param off starting offset into output buffer
287     * @return the number of encoded bytes written
288     * @throws ArrayIndexOutOfBoundsException if {@code off} is negative or the encoded value exceeds the given buffer
289     */
290    private static int encode(long value, byte[] buf, int off) {
291        int len = 1;
292        if (value < MIN_SINGLE_BYTE_VALUE) {
293            value += NEGATIVE_ADJUST;
294            long mask = 0x00ffffffffffffffL;
295            for (int shift = 56; shift != 0; shift -= 8, mask >>= 8) {
296                if ((value | mask) != ~0L)
297                    buf[off + len++] = (byte)(value >> shift);
298            }
299            buf[off] = (byte)(MIN_SINGLE_BYTE_ENCODED - len);
300        } else if (value > MAX_SINGLE_BYTE_VALUE) {
301            value += POSITIVE_ADJUST;
302            long mask = 0xff00000000000000L;
303            for (int shift = 56; shift != 0; shift -= 8, mask >>= 8) {
304                if ((value & mask) != 0L)
305                    buf[off + len++] = (byte)(value >> shift);
306            }
307            buf[off] = (byte)(MAX_SINGLE_BYTE_ENCODED + len);
308        } else {
309            buf[off] = (byte)(value + ZERO_ADJUST);
310            return 1;
311        }
312        buf[off + len++] = (byte)value;
313        return len;
314    }

316    /**
317     * Test routine.
318     *
319     * @param args command line arguments
320     */
321    public static void main(String[] args) {
322        for (String arg : args) {
323            byte[] bytes = null;
324            try {
325                bytes = ByteUtil.parse(arg);
326            } catch (IllegalArgumentException e) {
327                if (arg.startsWith("0x"))
328                    bytes = ByteUtil.parse(arg.substring(2));
329            }
330            if (bytes != null) {
331                final long value = LongEncoder.decode(bytes);
332                System.out.println("0x" + ByteUtil.toString(bytes)
333                  + " decodes to " + value + " (" + String.format("0x%016x", value) + ")");
334            }
335            Long value = null;
336            try {
337                value = Long.parseLong(arg);
338            } catch (IllegalArgumentException e) {
339                // ignore
340            }
341            if (value != null) {
342                final ByteWriter writer = new ByteWriter();
343                LongEncoder.write(writer, value);
344                System.out.println(value + " (" + String.format("0x%016x", value)
345                  + ") encodes to " + ByteUtil.toString(writer.getBytes()));
346            }
347        }
348    }
