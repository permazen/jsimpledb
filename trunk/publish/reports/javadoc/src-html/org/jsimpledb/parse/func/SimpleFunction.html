008package org.jsimpledb.parse.func;
009
010import org.jsimpledb.parse.ParseContext;
011import org.jsimpledb.parse.ParseSession;
012import org.jsimpledb.parse.expr.Node;
013import org.jsimpledb.parse.expr.Value;
014
015/**
016 * Simplified {@link Function} implementation for when the parameters are all normal expressions.
017 */
018public abstract class SimpleFunction extends AbstractFunction {
019
020    protected final int minArgs;
021    protected final int maxArgs;
022
023// Constructors
024
025    /**
026     * Constructor.
027     *
028     * @param name function name
029     * @param minArgs minimum number of arguments (inclusive)
030     * @param maxArgs maximum number of arguments (inclusive)
031     */
032    protected SimpleFunction(String name, int minArgs, int maxArgs) {
033        super(name);
034        if (minArgs < 0 || minArgs > maxArgs)
035            throw new IllegalArgumentException("invalid minArgs = " + minArgs + ", maxArgs = " + maxArgs);
036        this.minArgs = minArgs;
037        this.maxArgs = maxArgs;
038    }
039
040// Accessors
041
042    /**
043     * Get the minimum number of arguments allowed (inclusive).
044     *
045     * @return minimum required function arguments
046     */
047    public int getMinArgs() {
048        return this.minArgs;
049    }
050
051    /**
052     * Get the maximum number of arguments allowed (inclusive).
053     *
054     * @return maximum allowed function arguments
055     */
056    public int getMaxArgs() {
057        return this.maxArgs;
058    }
059
060// Parsing
061
062    @Override
063    public final Node[] parseParams(ParseSession session, ParseContext ctx, boolean complete) {
064        return this.parseExpressionParams(session, ctx, complete, 0, this.minArgs, this.maxArgs);
065    }
066
067    @Override
068    public final Value apply(ParseSession session, Object info) {
069        final Node[] params = (Node[])info;
070        final Value[] values = new Value[params.length];
071        for (int i = 0; i < params.length; i++)
072            values[i] = params[i].evaluate(session);
073        return this.apply(session, values);
074    }
075
076    /**
077     * Apply this function to the given values.
078     *
079     * @param session parse session
080     * @param params parsed parameters; will already be checked between {@link #getMinArgs} and {@link #getMaxArgs}
081     * @return value returned by this function
082     * @throws RuntimeException if there is an error
083     */
084    protected abstract Value apply(ParseSession session, Value[] params);
085}
