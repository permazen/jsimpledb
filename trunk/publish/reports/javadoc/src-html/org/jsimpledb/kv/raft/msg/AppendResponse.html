008package org.jsimpledb.kv.raft.msg;
009
010import com.google.common.base.Preconditions;
011
012import java.nio.ByteBuffer;
013
014import org.jsimpledb.kv.raft.Timestamp;
015import org.jsimpledb.util.LongEncoder;
016
017/**
018 * Sent from a follower to a leader in response to an {@link AppendRequest}.
019 *
020 * <p>
021 * Also contains information about the follower's log.
022 */
023public class AppendResponse extends Message {
024
025    private final Timestamp leaderTimestamp;        // leaderTimestamp from corresponding AppendRequest
026    private final boolean success;                  // true if previous log entry term and index matched
027    private final long matchIndex;                  // index of higest log entry known to match leader
028    private final long lastLogIndex;                // the index of the last log entry in follower's log
029
030// Constructors
031
032    /**
033     * Constructor.
034     *
035     * @param clusterId cluster ID
036     * @param senderId identity of sender
037     * @param recipientId identity of recipient
038     * @param term sender's current term
039     * @param leaderTimestamp copy of {@link AppendRequest#getLeaderTimestamp}
040     * @param success if request successfully matched
041     * @param matchIndex highest known matching log index
042     * @param lastLogIndex index of follower's last log entry
043     */
044    public AppendResponse(int clusterId, String senderId, String recipientId, long term, Timestamp leaderTimestamp,
045      boolean success, long matchIndex, long lastLogIndex) {
046        super(Message.APPEND_RESPONSE_TYPE, clusterId, senderId, recipientId, term);
047        this.leaderTimestamp = leaderTimestamp;
048        this.success = success;
049        this.matchIndex = matchIndex;
050        this.lastLogIndex = lastLogIndex;
051        this.checkArguments();
052    }
053
054    AppendResponse(ByteBuffer buf) {
055        super(Message.APPEND_RESPONSE_TYPE, buf);
056        this.leaderTimestamp = Message.getTimestamp(buf);
057        this.success = Message.getBoolean(buf);
058        this.matchIndex = LongEncoder.read(buf);
059        this.lastLogIndex = LongEncoder.read(buf);
060        this.checkArguments();
061    }
062
063    @Override
064    void checkArguments() {
065        super.checkArguments();
066        Preconditions.checkArgument(this.leaderTimestamp != null);
067        Preconditions.checkArgument(this.matchIndex >= -1);
068        Preconditions.checkArgument(this.matchIndex <= this.lastLogIndex);
069    }
070
071// Properties
072
073    /**
074     * Get the {@code leaderTimestamp} from the corresponding {@link AppendRequest}.
075     *
076     * @return request leader timestamp
077     * @see AppendRequest#getLeaderTimestamp
078     */
079    public Timestamp getLeaderTimestamp() {
080        return this.leaderTimestamp;
081    }
082
083    /**
084     * Determine whether the request was successful, i.e., the previous log entry term and index matched.
085     *
086     * @return true if the corresponding request matched
087     */
088    public boolean isSuccess() {
089        return this.success;
090    }
091
092    /**
093     * Get the index of the most recent log entry known to match leader.
094     * In case of a successful request, this will be equal to the log entry sent (or the previous
095     * log entry in case of a probe). In case of a failed request, this will be equal
096     * to the follower's state machine last applied index.
097     *
098     * @return highest known matching log entry index
099     */
100    public long getMatchIndex() {
101        return this.matchIndex;
102    }
103
104    /**
105     * Get the index of the last log entry in the follower's log.
106     *
107     * @return last log entry index
108     */
109    public long getLastLogIndex() {
110        return this.lastLogIndex;
111    }
112
113// Message
114
115    @Override
116    public void visit(MessageSwitch handler) {
117        handler.caseAppendResponse(this);
118    }
119
120    @Override
121    public void writeTo(ByteBuffer dest) {
122        super.writeTo(dest);
123        Message.putTimestamp(dest, this.leaderTimestamp);
124        Message.putBoolean(dest, this.success);
125        LongEncoder.write(dest, this.matchIndex);
126        LongEncoder.write(dest, this.lastLogIndex);
127    }
128
129    @Override
130    protected int calculateSize() {
131        return super.calculateSize()
132          + Message.calculateSize(this.leaderTimestamp)
133          + 1
134          + LongEncoder.encodeLength(this.matchIndex)
135          + LongEncoder.encodeLength(this.lastLogIndex);
136    }
137
138// Object
139
140    @Override
141    public String toString() {
142        return this.getClass().getSimpleName()
143          + "[\"" + this.getSenderId() + "\"->\"" + this.getRecipientId() + "\""
144          + ",clusterId=" + String.format("%08x", this.getClusterId())
145          + ",term=" + this.getTerm()
146          + ",leaderTimestamp=" + this.leaderTimestamp
147          + ",success=" + this.success
148          + ",matchIndex=" + this.matchIndex
149          + ",lastLogIndex=" + this.lastLogIndex
150          + "]";
151    }
152}
