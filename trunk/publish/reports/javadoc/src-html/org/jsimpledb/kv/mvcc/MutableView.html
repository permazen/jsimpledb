027/**
028 * Provides a mutable view of an underlying, read-only {@link KVStore}.
029 *
030 * <p>
031 * Instances intercept all operations to the underlying {@link KVStore}, recording mutations in a {@link Writes} instance
032 * instead of applying them to the {@link KVStore}. Instances then provide a view of the mutated {@link KVStore} based those
033 * mutations. Mutations that overwrite previous mutations are consolidated.
034 * </p>
035 *
036 * <p>
037 * Unlike writes, reads are passed through to the underlying {@link KVStore}, except where they intersect a previous write.
038 * Reads may also be optionally recorded.
039 * </p>
040 *
041 * <p>
042 * In all cases, then underlying {@link KVStore} is never modified.
043 * </p>
044 *
045 * <p>
046 * Instances are also capable of performing certain MVCC-related calculations, such as whether two transactions may be re-ordered.
047 * </p>
048 *
049 * <p>
050 * Instances are thread safe; however, directly accessing the associated {@link Reads} or {@link Writes} is not thread safe.
051 * </p>
052 */
053public class MutableView extends AbstractKVStore implements SizeEstimating {
054
055    private final KVStore kv;
056    private final Writes writes;
057    private Reads reads;
058
059// Constructors
060
061    /**
062     * Constructor.
063     *
064     * @param kv underlying {@link KVStore}
065     * @throws IllegalArgumentException if {@code kv} is null
066     */
067    public MutableView(KVStore kv) {
068        this(kv, new Reads(), new Writes());
069    }
070
071    /**
072     * Constructor using caller-provided {@link Reads} (optional} and {@link Writes}.
073     *
074     * @param kv underlying {@link KVStore}
075     * @param reads recorded reads, or null for none
076     * @param writes recorded writes
077     * @throws IllegalArgumentException if {@code kv} is null
078     * @throws IllegalArgumentException if {@code writes} is null
079     */
080    public MutableView(KVStore kv, Reads reads, Writes writes) {
081        Preconditions.checkArgument(kv != null, "null kv");
082        Preconditions.checkArgument(writes != null, "null writes");
083        this.kv = kv;
084        this.reads = reads;
085        this.writes = writes;
086        synchronized (this) { }                                     // because this.reads is not final
087    }

123// KVStore

125    @Override
126    public synchronized byte[] get(byte[] key) {
127
128        // Sanity check
129        assert this.check();
130
131        // Check puts
132        final byte[] putValue = this.writes.getPuts().get(key);
133        if (putValue != null)
134            return putValue;
135
136        // Check removes
137        if (this.writes.getRemoves().contains(key))
138            return null;
139
140        // Read from k/v store
141        byte[] value = this.kv.get(key);
142
143        // Record the read
144        this.recordReads(key, ByteUtil.getNextKey(key));
145
146        // Check counter adjustments
147        if (value == null)                      // we can ignore adjustments of missing values
148            return null;
149        final Long adjust = this.writes.getAdjusts().get(key);
150        if (adjust != null) {
151
152            // Decode value we just read as a counter
153            final long counterValue;
154            try {
155                counterValue = this.kv.decodeCounter(value);
156            } catch (IllegalArgumentException e) {
157                this.writes.getAdjusts().remove(key);       // previous adjustment was bogus because value was not decodable
158                return value;
159            }
160
161            // Adjust counter value by adjustment and re-encode
162            value = this.kv.encodeCounter(counterValue + adjust);
163        }
164
165        // Done
166        return value;
167    }

169    @Override
170    public Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, boolean reverse) {
171
172        // Sanity check
173        assert this.check();
174
175        // Build iterator
176        return new RangeIterator(minKey, maxKey, reverse);
177    }
178
179    @Override
180    public synchronized void put(byte[] key, byte[] value) {
181
182        // Sanity check
183        assert this.check();
184        Preconditions.checkArgument(key != null, "null key");
185        Preconditions.checkArgument(value != null, "null value");
186
187        // Overwrite any counter adjustment
188        this.writes.getAdjusts().remove(key);
189
190        // Overwrite any removal
191        if (this.writes.getRemoves().contains(key))
192            this.writes.setRemoves(this.writes.getRemoves().remove(new KeyRange(key)));
193
194        // Record the put
195        this.writes.getPuts().put(key.clone(), value.clone());
196    }
197
198    @Override
199    public synchronized void remove(byte[] key) {
200
201        // Sanity check
202        assert this.check();
203        Preconditions.checkArgument(key != null, "null key");
204
205        // Overwrite any counter adjustment
206        this.writes.getAdjusts().remove(key);
207
208        // Overwrite any put
209        this.writes.getPuts().remove(key);
210
211        // Record the remove
212        this.writes.setRemoves(this.writes.getRemoves().add(new KeyRange(key)));
213    }
214
215    @Override
216    public synchronized void removeRange(byte[] minKey, byte[] maxKey) {
217
218        // Sanity check
219        assert this.check();
220
221        // Realize minKey
222        if (minKey == null)
223            minKey = ByteUtil.EMPTY;
224
225        // Overwrite any puts and counter adjustments
226        if (maxKey != null) {
227            this.writes.getPuts().subMap(minKey, maxKey).clear();
228            this.writes.getAdjusts().subMap(minKey, maxKey).clear();
229        } else {
230            this.writes.getPuts().tailMap(minKey).clear();
231            this.writes.getAdjusts().tailMap(minKey).clear();
232        }
233
234        // Record the remove
235        this.writes.setRemoves(this.writes.getRemoves().add(new KeyRange(minKey, maxKey)));
236    }

248    @Override
249    public synchronized void adjustCounter(byte[] key, long amount) {
250
251        // Sanity check
252        assert this.check();
253
254        // Check puts
255        final byte[] putValue = this.writes.getPuts().get(key);
256        if (putValue != null) {
257            final long value;
258            try {
259                value = this.kv.decodeCounter(putValue);
260            } catch (IllegalArgumentException e) {
261                return;                                 // previously put value was not decodable, so ignore this adjustment
262            }
263            this.writes.getPuts().put(key, this.kv.encodeCounter(value + amount));
264            return;
265        }
266
267        // Check removes
268        if (this.writes.getRemoves().contains(key))
269            return;
270
271        // Calculate new, cumulative adjustment
272        final Long oldAdjust = this.writes.getAdjusts().get(key);
273        final long adjust = (oldAdjust != null ? oldAdjust : 0) + amount;
274
275        // Record/update adjustment
276        if (adjust != 0)
277            this.writes.getAdjusts().put(key, adjust);
278        else
279            this.writes.getAdjusts().remove(key);
280    }

311// Internal methods

313    // Record that keys were read in the range [minKey, maxKey)
314    private synchronized void recordReads(byte[] minKey, byte[] maxKey) {
315
316        // Not tracking reads?
317        if (this.reads == null)
318            return;
319
320        // Realize minKey
321        if (minKey == null)
322            minKey = ByteUtil.EMPTY;
323
324        // Already tracked?
325        final KeyRange readRange = new KeyRange(minKey, maxKey);
326        if (this.reads.getReads().contains(readRange))
327            return;
328
329        // Subtract out the part of the read range that did not really go through to k/v store due to puts or removes
330        KeyRanges readRanges = new KeyRanges(readRange);
331        final Set<byte[]> putKeys = (maxKey != null ?
332          this.writes.getPuts().subMap(minKey, maxKey) : this.writes.getPuts().tailMap(minKey)).keySet();
333        for (byte[] key : putKeys)
334            readRanges = readRanges.remove(new KeyRange(key));
335        readRanges = readRanges.intersection(this.writes.getRemoves().inverse());
336
337        // Record reads
338        if (!readRanges.isEmpty())
339            this.reads.setReads(this.reads.getReads().union(readRanges));
340    }

342// Debugging
343
344    // Verify puts, removes, and adjusts are all mutually disjoint
345    private synchronized boolean check() {
346        MutableView.verifyDisjoint(
347          this.writes.getRemoves(),
348          MutableView.buildKeyRanges(this.writes.getPuts().keySet()),
349          MutableView.buildKeyRanges(this.writes.getAdjusts().keySet()));
350        return true;
351    }
352
353    private static void verifyDisjoint(KeyRanges... rangeses) {
354        for (int i = 0; i < rangeses.length - 1; i++) {
355            for (int j = i + 1; j < rangeses.length; j++)
356                assert rangeses[i].intersection(rangeses[j]).isEmpty();
357        }
358    }
359
360    private static KeyRanges buildKeyRanges(Iterable<byte[]> keys) {
361        KeyRanges ranges = KeyRanges.EMPTY;
362        for (byte[] key : keys)
363            ranges = ranges.add(new KeyRange(key));
364        return ranges;
365    }

369    private class RangeIterator implements Iterator<KVPair> {
370
371        private final KVPairIterator pi;
372        private final byte[] limit;
373        private final boolean reverse;
374
375        private byte[] cursor;                  // current position; inclusive if forward, exclusive if reverse
376        private KVPair next;                    // the next k/v pair queued up, or null if not found yet
377        private byte[] removeKey;               // key to remove if remove() is invoked
378        private boolean finished;
379
380        RangeIterator(byte[] minKey, byte[] maxKey, boolean reverse) {
381
382            // Realize minKey
383            if (minKey == null)
384                minKey = ByteUtil.EMPTY;
385
386            // Build KVPairIterator that omits keys we've put or removed so far; this is safe even if more keys are put and/or
387            // removed after creation, because the set "keys we've put or removed so far" can only increase over time.
388            synchronized (MutableView.this) {
389                final KeyRanges putsAndRemoves = MutableView.buildKeyRanges(MutableView.this.writes.getPuts().keySet())
390                  .union(MutableView.this.writes.getRemoves());
391                this.pi = new KVPairIterator(MutableView.this.kv, new KeyRange(minKey, maxKey), putsAndRemoves.inverse(), reverse);
392            }
393
394            // Initialize cursor
395            this.cursor = reverse ? maxKey : minKey;
396            this.limit = reverse ? minKey : maxKey;
397            this.reverse = reverse;
398        }
399
400        @Override
401        public synchronized boolean hasNext() {
402            return this.next != null || this.findNext();
403        }
404
405        @Override
406        public synchronized KVPair next() {
407            if (this.next == null && !this.findNext())
408                throw new NoSuchElementException();
409            final KVPair pair = this.next;
410            assert pair != null;
411            this.removeKey = pair.getKey();
412            this.next = null;
413            return pair;
414        }
415
416        @Override
417        public synchronized void remove() {
418            Preconditions.checkState(this.removeKey != null);
419            MutableView.this.remove(this.removeKey);
420            this.removeKey = null;
421        }
422
423        private synchronized boolean findNext() {
424
425            // Invariants
426            assert this.next == null;
427            assert this.cursor != null || this.reverse;
428
429            // Exhausted?
430            if (this.finished)
431                return false;
432
433            // Compare against most current MutableView puts and removes
434            synchronized (MutableView.this) {
435
436                // Find the next k/v pair not filtered out by (newly added) removes, if any
437                KVPair readPair = null;
438                while (this.pi.hasNext()) {
439                    readPair = this.pi.next();
440                    if (MutableView.this.writes.getRemoves().contains(readPair.getKey())) {
441                        readPair = null;
442                        continue;
443                    }
444                    break;
445                }
446
447                // Find the next put, if any
448                final Map.Entry<byte[], byte[]> putEntry = this.reverse ?
449                  (this.cursor != null ?
450                   MutableView.this.writes.getPuts().lowerEntry(this.cursor) : MutableView.this.writes.getPuts().lastEntry()) :
451                  MutableView.this.writes.getPuts().ceilingEntry(this.cursor);
452                final KVPair putPair = putEntry != null ? new KVPair(putEntry) : null;
453
454                // Figure out which pair wins (read or put)
455                final KVPair pair;
456                int diff = 0;
457                if (readPair == null && putPair == null)
458                    pair = null;
459                else if (readPair == null)
460                    pair = putPair;
461                else if (putPair == null)
462                    pair = readPair;
463                else {
464                    diff = ByteUtil.compare(putPair.getKey(), readPair.getKey());
465                    if (reverse)
466                        diff = -diff;
467                    pair = diff <= 0 ? putPair : readPair;                  // if there's a tie, the put wins
468                }
469
470                // Record that we read from everything we just scanned over in the underlying KVStore
471                final byte[] skipMin;
472                final byte[] skipMax;
473                if (this.reverse) {
474                    skipMin = pair != null ? pair.getKey() : ByteUtil.EMPTY;
475                    skipMax = this.cursor;
476                } else {
477                    skipMin = this.cursor;
478                    skipMax = pair != null ? ByteUtil.getNextKey(pair.getKey()) : null;
479                }
480                if (skipMax == null || ByteUtil.compare(skipMin, skipMax) < 0)
481                    MutableView.this.recordReads(skipMin, skipMax);
482
483                // Finished?
484                if (pair == null) {
485                    this.finished = true;
486                    return false;
487                }
488
489                // Update state
490                this.next = pair;
491                this.cursor = pair != null ? (this.reverse ? pair.getKey() : ByteUtil.getNextKey(pair.getKey())) : null;
492
493                // If a put appeared prior to the KVPPairIterator's next k/v pair, backup the KVPairIterator for next time
494                if (diff < 0)
495                    this.pi.setNextTarget(this.cursor);
496
497                // Done
498                return this.next != null;
499            }
500        }
501    }
