008package org.jsimpledb.kv.mvcc;
009
010import org.jsimpledb.kv.CloseableKVStore;
011import org.jsimpledb.kv.KVStore;
012
013/**
014 * Extension of the {@link KVStore} interface for implementations that support atomic, batched reads and writes.
015 *
016 * <p>
017 * Atomic batch reads are available via {@link #snapshot}, which returns a consistent point-in-time views of the {@link KVStore}.
018 * Atomic batch writes are available via {@link #mutate mutate()}, which applies a set of mutations in an "all or none" fashion.
019 */
020public interface AtomicKVStore extends KVStore {
021
022    /**
023     * Acquire a read-only, snapshot view of this instance based on the current state.
024     *
025     * <p>
026     * The returned {@link KVStore} view should remain constant even if this instance is subsequently mutated.
027     *
028     * <p>
029     * Note: callers are required to {@link CloseableKVStore#close close} the returned instance when no longer in use.
030     *
031     * @return read-only, snapshot view of this instance
032     * @throws org.jsimpledb.kv.StaleTransactionException if an underlying transaction is no longer usable
033     * @throws org.jsimpledb.kv.RetryTransactionException if an underlying transaction must be retried and is no longer usable
034     */
035    CloseableKVStore snapshot();
036
037    /**
038     * Apply a set of mutations to this instance atomically.
039     *
040     * <p>
041     * If this method returns normally, all of the given mutations will have been applied. If this method
042     * returns abnormally, then none of the given mutations will have been applied.
043     *
044     * <p>
045     * In any case, other threads observing this instance will never see a partial application of the given mutations.
046     *
047     * <p>
048     * This method is required to apply the mutations in this order: removes, puts, adjusts.
049     *
050     * <p>
051     * If {@code sync} is true, the implementation must durably persist the changes before returning.
052     *
053     * @param mutations the mutations to apply
054     * @param sync if true, caller requires that the changes be durably persisted
055     * @throws org.jsimpledb.kv.StaleTransactionException if an underlying transaction is no longer usable
056     * @throws org.jsimpledb.kv.RetryTransactionException if an underlying transaction must be retried and is no longer usable
057     * @throws UnsupportedOperationException if {@code sync} is true and this implementation cannot guarantee durability
058     * @throws IllegalArgumentException if {@code mutations} is null
059     */
060    void mutate(Mutations mutations, boolean sync);
061}
