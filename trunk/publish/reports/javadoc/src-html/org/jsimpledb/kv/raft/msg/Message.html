028/**
029 * Support superclass for Raft messages.
030 */
031public abstract class Message {
032
033    // Codes identifying serialized message type
034    static final byte APPEND_REQUEST_TYPE = 1;
035    static final byte APPEND_RESPONSE_TYPE = 2;
036    static final byte COMMIT_REQUEST_TYPE = 3;
037    static final byte COMMIT_RESPONSE_TYPE = 4;
038    static final byte GRANT_VOTE_TYPE = 5;
039    static final byte INSTALL_SNAPSHOT_TYPE = 6;
040    static final byte REQUEST_VOTE_TYPE = 7;
041
042    // Serialization version number
043    private static final byte VERSION_1 = 1;
044
045    // Minimum buffer size to use a direct buffer
046    private static final int MIN_DIRECT_BUFFER_SIZE = 128;
047
048    private final byte type;
049    private final int clusterId;
050    private final String senderId;
051    private final String recipientId;
052    private final long term;
053
054    protected Message(byte type, int clusterId, String senderId, String recipientId, long term) {
055        this.type = type;
056        this.clusterId = clusterId;
057        this.senderId = senderId;
058        this.recipientId = recipientId;
059        this.term = term;
060    }
061
062    protected Message(byte type, ByteBuffer buf) {
063        this.type = type;
064        this.clusterId = buf.getInt();
065        this.senderId = Message.getString(buf);
066        this.recipientId = Message.getString(buf);
067        this.term = LongEncoder.read(buf);
068    }
069
070    void checkArguments() {
071        Preconditions.checkArgument(this.type >= APPEND_REQUEST_TYPE && this.type <= REQUEST_VOTE_TYPE);
072        Preconditions.checkArgument(this.clusterId != 0);
073        Preconditions.checkArgument(this.senderId != null);
074        Preconditions.checkArgument(this.recipientId != null);
075        Preconditions.checkArgument(this.term > 0);
076    }

134// (De)serialization methods

136    /**
137     * Decode a message from the given input.
138     *
139     * <p>
140     * Note that data is not necessarily copied out of {@code buf}, so the returned instance may become invalid
141     * if the data in {@code buf} gets overwritten.
142     *
143     * @param buf source for encoded message
144     * @return decoded message
145     * @throws java.nio.BufferUnderflowException if there is not enough data
146     * @throws IllegalArgumentException if encoded message is bogus
147     * @throws IllegalArgumentException if there is trailing garbage
148     */
149    public static Message decode(ByteBuffer buf) {
150
151        // Check encoding format version
152        final byte version = buf.get();
153        switch (version) {
154        case Message.VERSION_1:
155            break;
156        default:
157            throw new IllegalArgumentException("unrecognized message format version " + version);
158        }
159
160        // Read type and decode message
161        final Message msg;
162        final byte type = buf.get();
163        switch (type) {
164        case APPEND_REQUEST_TYPE:
165            msg = new AppendRequest(buf);
166            break;
167        case APPEND_RESPONSE_TYPE:
168            msg = new AppendResponse(buf);
169            break;
170        case COMMIT_REQUEST_TYPE:
171            msg = new CommitRequest(buf);
172            break;
173        case COMMIT_RESPONSE_TYPE:
174            msg = new CommitResponse(buf);
175            break;
176        case GRANT_VOTE_TYPE:
177            msg = new GrantVote(buf);
178            break;
179        case INSTALL_SNAPSHOT_TYPE:
180            msg = new InstallSnapshot(buf);
181            break;
182        case REQUEST_VOTE_TYPE:
183            msg = new RequestVote(buf);
184            break;
185        default:
186            throw new IllegalArgumentException("invalid message type " + type);
187        }
188
189        // Check for trailing garbage
190        if (buf.hasRemaining())
191            throw new IllegalArgumentException("buffer contains " + buf.remaining() + " bytes of extra garbage after " + msg);
192
193        // Done
194        return msg;
195    }
196
197    /**
198     * Serialize this instance.
199     *
200     * @return encoded message
201     */
202    public ByteBuffer encode() {
203        final int size = this.calculateSize();
204        final ByteBuffer buf = size >= MIN_DIRECT_BUFFER_SIZE ? ByteBuffer.allocateDirect(size) : ByteBuffer.allocate(size);
205        this.writeTo(buf);
206        if (buf.hasRemaining())
207            throw new RuntimeException("internal error: " + buf.remaining() + " remaining bytes in buffer from " + this);
208        return (ByteBuffer)buf.flip();
209    }
210
211    /**
212     * Serialize this instance into the given buffer.
213     *
214     * @param buf destination for encoded data
215     * @throws java.nio.BufferOverflowException if data overflows {@code buf}
216     */
217    public void writeTo(ByteBuffer buf) {
218        buf.put(Message.VERSION_1);
219        buf.put(this.type);
220        buf.putInt(this.clusterId);
221        Message.putString(buf, this.senderId);
222        Message.putString(buf, this.recipientId);
223        LongEncoder.write(buf, this.term);
224    }
225
226    /**
227     * Calculate an upper bound on the number of bytes required by {@link #writeTo writeTo()}.
228     *
229     * @return an upper bound on the number of encoded bytes
230     */
231    protected int calculateSize() {
232        return 1
233          + 1
234          + 4
235          + Message.calculateSize(this.senderId)
236          + Message.calculateSize(this.recipientId)
237          + LongEncoder.encodeLength(this.term);
238    }

245// Helpers

247    /**
248     * Serialize a {@link ByteBuffer} into the buffer.
249     *
250     * @param dest destination for encoded data
251     * @param buf data to encode
252     * @throws ReadOnlyBufferException if {@code dest} is read only
253     * @throws BufferOverflowException if {@code dest} overflows
254     * @throws IllegalArgumentException if {@code buf} has more than 2<sup>^31</sup> bytes remaining
255     * @throws IllegalArgumentException if either parameter is null
256     */
257    protected static void putByteBuffer(ByteBuffer dest, ByteBuffer buf) {
258        Preconditions.checkArgument(dest != null, "null dest");
259        Preconditions.checkArgument(buf != null, "null buf");
260        UnsignedIntEncoder.write(dest, buf.remaining());
261        dest.put(buf.asReadOnlyBuffer());
262    }
263
264    /**
265     * Deserialize a {@link ByteBuffer} previously serialized by {@link #putByteBuffer putByteBuffer()} from the buffer.
266     *
267     * @param buf source for encoded data
268     * @return decoded data
269     * @throws BufferUnderflowException if {@code buf} underflows
270     * @throws IllegalArgumentException if input is bogus
271     * @throws IllegalArgumentException if {@code buf} is null
272     */
273    protected static ByteBuffer getByteBuffer(ByteBuffer buf) {
274        Preconditions.checkArgument(buf != null, "null buf");
275        final int numBytes = UnsignedIntEncoder.read(buf);
276        if (numBytes > buf.remaining())
277            throw new IllegalArgumentException("bogus buffer length " + numBytes + " > " + buf.remaining());
278        final ByteBuffer result = (ByteBuffer)buf.slice().limit(numBytes);
279        buf.position(buf.position() + numBytes);
280        return result;
281    }
282
283    protected static int calculateSize(ByteBuffer buf) {
284        return UnsignedIntEncoder.encodeLength(buf.remaining()) + buf.remaining();
285    }
286
287    /**
288     * Serialize a {@link String} into the buffer.
289     *
290     * @param dest destination for encoded data
291     * @param string string to encode
292     * @throws ReadOnlyBufferException if {@code dest} is read only
293     * @throws BufferOverflowException if {@code dest} overflows
294     * @throws IllegalArgumentException if either parameter is null
295     */
296    protected static void putString(ByteBuffer dest, String string) {
297        Preconditions.checkArgument(dest != null, "null dest");
298        Preconditions.checkArgument(string != null, "null string");
299        for (int i = 0; i < string.length(); i++) {
300            final int c = string.charAt(i);
301            if (c >= 0x0001 && c <= 0x007f)
302                dest.put((byte)c);
303            else if (c < 0x0800) {
304                dest.put((byte)(0xc0 | ((c >> 6) & 0x1f)));
305                dest.put((byte)(0x80 | ((c >> 0) & 0x3f)));
306            } else {
307                dest.put((byte)(0xe0 | ((c >> 12) & 0x0f)));
308                dest.put((byte)(0x80 | ((c >>  6) & 0x3f)));
309                dest.put((byte)(0x80 | ((c >>  0) & 0x3f)));
310            }
311        }
312        dest.put((byte)0);
313    }
314
315    /**
316     * Deserialize a {@link String} previously serialized by {@link #putString putString()} from the buffer.
317     *
318     * @param buf source for encoded data
319     * @return decoded string, never null
320     * @throws BufferUnderflowException if {@code buf} underflows
321     * @throws IllegalArgumentException if input is bogus
322     */
323    protected static String getString(ByteBuffer buf) {
324        Preconditions.checkArgument(buf != null, "null buf");
325        final StringWriter writer = new StringWriter();
326        while (true) {
327            final int b1 = buf.get() & 0xff;
328            if (b1 == 0)
329                break;
330            if ((b1 & 0x80) == 0)
331                writer.append((char)b1);
332            else if ((b1 & 0xe0) == 0xc0) {
333                final int b2 = buf.get() & 0xff;
334                if ((b2 & 0xc0) != 0x80)
335                    throw new IllegalArgumentException("invalid UTF-8 sequence: " + b1 + " " + b2);
336                writer.append((char)((b1 & 0x1f) << 6 | (b2 & 0x3f)));
337            } else if ((b1 & 0xf0) == 0xe0) {
338                final int b2 = buf.get() & 0xff;
339                final int b3 = buf.get() & 0xff;
340                if ((b2 & 0xc0) != 0x80 || (b3 & 0xc0) != 0x80)
341                    throw new IllegalArgumentException("invalid UTF-8 sequence: " + b1 + " " + b2 + " " + b3);
342                writer.append((char)(((b1 & 0x0f) << 12) | ((b2 & 0x3f) << 6) | (b3 & 0x3f)));
343            } else
344                throw new IllegalArgumentException("invalid UTF-8 sequence: " + b1);
345        }
346        return writer.toString();
347    }
348
349    protected static int calculateSize(String string) {
350        Preconditions.checkArgument(string != null, "null string");
351        int total = 1;
352        for (int i = 0; i < string.length(); i++) {
353            final int ch = string.charAt(i);
354            total += (ch != 0x0000 && ch < 0x0080) ? 1 : ch < 0x0800 ? 2 : 3;
355        }
356        return total;
357    }

359    /**
360     * Serialize a boolean value into the buffer.
361     *
362     * @param dest destination for encoded data
363     * @param value value to encode
364     * @throws ReadOnlyBufferException if {@code dest} is read only
365     * @throws BufferOverflowException if {@code dest} overflows
366     * @throws IllegalArgumentException if {@code dest} is null
367     */
368    protected static void putBoolean(ByteBuffer dest, boolean value) {
369        Preconditions.checkArgument(dest != null, "null dest");
370        dest.put(value ? (byte)1 : (byte)0);
371    }

373    /**
374     * Deserialize a boolean value previously serialized by {@link #putBoolean putBoolean()} from the buffer.
375     *
376     * @param buf source for encoded data
377     * @return decoded value
378     * @throws BufferUnderflowException if {@code buf} underflows
379     * @throws IllegalArgumentException if input is bogus
380     */
381    protected static boolean getBoolean(ByteBuffer buf) {
382        Preconditions.checkArgument(buf != null, "null buf");
383        switch (buf.get()) {
384        case (byte)0:
385            return false;
386        case (byte)1:
387            return true;
388        default:
389            throw new IllegalArgumentException("read invalid boolean value");
390        }
391    }

425    /**
426     * Serialize a {@link Reads} into the buffer.
427     *
428     * @param dest destination for encoded data
429     * @param reads value to encode
430     * @throws ReadOnlyBufferException if {@code dest} is read only
431     * @throws BufferOverflowException if {@code dest} overflows
432     * @throws IllegalArgumentException if either parameter is null
433     */
434    protected static void putReads(ByteBuffer dest, Reads reads) {
435        Preconditions.checkArgument(dest != null, "null dest");
436        Preconditions.checkArgument(reads != null, "null reads");
437        final ByteBufferOutputStream output = new ByteBufferOutputStream(dest);
438        try {
439            reads.serialize(output);
440            output.flush();
441        } catch (IOException e) {
442            if (e.getCause() instanceof BufferOverflowException)
443                throw (BufferOverflowException)e.getCause();
444            if (e.getCause() instanceof ReadOnlyBufferException)
445                throw (ReadOnlyBufferException)e.getCause();
446            throw new RuntimeException("unexpected exception", e);
447        }
448    }

513// Debugging

515    String describe(ByteBuffer buf) {
516        if (buf == null)
517            return null;
518        final int size = buf.remaining();
519        if (size <= 32) {
520            final byte[] data = new byte[size];
521            buf.asReadOnlyBuffer().get(data);
522            return ByteUtil.toString(data);
523        }
524        return size + " bytes";
525    }
