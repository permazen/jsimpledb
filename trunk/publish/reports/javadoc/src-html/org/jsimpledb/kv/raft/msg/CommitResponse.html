008package org.jsimpledb.kv.raft.msg;
009
010import com.google.common.base.Preconditions;
011
012import java.nio.ByteBuffer;
013
014import org.jsimpledb.kv.raft.Timestamp;
015import org.jsimpledb.util.LongEncoder;
016
017/**
018 * Response to a {@link CommitRequest}.
019 *
020 * <p>
021 * If the request was successful, the leader returns the term and index of the log entry that, when committed (in the Raft sense),
022 * allows the transaction to be committed. This suffices for read-write transactions, because the log entry corresponds
023 * directly to the transaction's mutations.
024 *
025 * <p>
026 * In order to support linearizable semantics for read-only transactions, which do not create new log entries, in addition
027 * a {@linkplain #getCommitLeaderLeaseTimeout minimum leader lease timeout} value is included; the transaction can then
028 * be committed after receiving an {@link AppendRequest} whose {@linkplain AppendRequest#getLeaderLeaseTimeout
029 * leader lease timeout} is at least this high. In most cases, such an {@link AppendRequest} will have already been received,
030 * so the transaction can be committed with a single round trip.
031 */
032public class CommitResponse extends Message {
033
034    private final long txId;
035    private final long commitTerm;
036    private final long commitIndex;
037    private final Timestamp commitLeaderLeaseTimeout;               // minimum required value we must see to commit
038    private final String errorMessage;
039
040// Constructors
041
042    /**
043     * Constructor for success case, when there is no minimum leader lease timeout required for commit
044     * (read-write transaction, or read-only transaction occurring within the current leader lease timeout).
045     *
046     * @param clusterId cluster ID
047     * @param senderId sending node identity
048     * @param recipientId identity of recipient
049     * @param term current term
050     * @param txId recipient's original transaction ID
051     * @param commitTerm transaction commit term
052     * @param commitIndex transaction commit index
053     */
054    public CommitResponse(int clusterId, String senderId, String recipientId, long term, long txId,
055      long commitTerm, long commitIndex) {
056        this(clusterId, senderId, recipientId, term, txId, commitTerm, commitIndex, null);
057    }
058
059    /**
060     * Constructor for success case when a minimum leader lease timeout is required for commit (read-only transaction
061     * when leader has not heard from a majority of followers in at least a minimum election timeout.
062     *
063     * @param clusterId cluster ID
064     * @param senderId sending node identity
065     * @param recipientId identity of recipient
066     * @param term current term
067     * @param txId recipient's original transaction ID
068     * @param commitTerm transaction commit term
069     * @param commitIndex transaction commit index
070     * @param commitLeaderLeaseTimeout minimum leader lease time required for commit, or null for none
071     */
072    public CommitResponse(int clusterId, String senderId, String recipientId, long term,
073      long txId, long commitTerm, long commitIndex, Timestamp commitLeaderLeaseTimeout) {
074        this(clusterId, senderId, recipientId, term, txId, commitTerm, commitIndex, commitLeaderLeaseTimeout, null);
075    }
076
077    /**
078     * Constructor for error case.
079     *
080     * @param clusterId cluster ID
081     * @param senderId sending node identity
082     * @param recipientId identity of recipient
083     * @param term current term
084     * @param txId recipient's original transaction ID
085     * @param errorMessage failure error message
086     */
087    public CommitResponse(int clusterId, String senderId, String recipientId, long term, long txId, String errorMessage) {
088        this(clusterId, senderId, recipientId, term, txId, 0, 0, null, errorMessage);
089        Preconditions.checkArgument(errorMessage != null, "null errorMessage");
090    }
091
092    private CommitResponse(int clusterId, String senderId, String recipientId, long term, long txId,
093      long commitTerm, long commitIndex, Timestamp commitLeaderLeaseTimeout, String errorMessage) {
094        super(Message.COMMIT_RESPONSE_TYPE, clusterId, senderId, recipientId, term);
095        this.txId = txId;
096        this.commitTerm = commitTerm;
097        this.commitIndex = commitIndex;
098        this.commitLeaderLeaseTimeout = commitLeaderLeaseTimeout;
099        this.errorMessage = errorMessage;
100        this.checkArguments();
101    }
102
103    CommitResponse(ByteBuffer buf) {
104        super(Message.COMMIT_RESPONSE_TYPE, buf);
105        this.txId = LongEncoder.read(buf);
106        this.commitTerm = LongEncoder.read(buf);
107        this.commitIndex = LongEncoder.read(buf);
108        this.commitLeaderLeaseTimeout = Message.getBoolean(buf) ? Message.getTimestamp(buf) : null;
109        this.errorMessage = Message.getBoolean(buf) ? Message.getString(buf) : null;
110        this.checkArguments();
111    }
112
113    @Override
114    void checkArguments() {
115        super.checkArguments();
116        Preconditions.checkArgument(this.txId != 0);
117        Preconditions.checkArgument(this.commitTerm >= 0);
118        Preconditions.checkArgument(this.commitIndex >= 0);
119        Preconditions.checkArgument(this.errorMessage == null || this.commitLeaderLeaseTimeout == null);
120    }
121
122// Properties
123
124    /**
125     * Get the sender's ID for the transaction.
126     *
127     * @return transaction ID
128     */
129    public long getTxId() {
130        return this.txId;
131    }
132
133    /**
134     * Determine whether the request was successful.
135     *
136     * @return true for success, false for error
137     */
138    public boolean isSuccess() {
139        return this.errorMessage == null;
140    }
141
142    /**
143     * Get the commit term for the transaction. This is always the same as the term of the sender.
144     *
145     * @return transaction commit term
146     */
147    public long getCommitTerm() {
148        return this.commitTerm;
149    }
150
151    /**
152     * Get the commit index for the transaction.
153     *
154     * @return transaction commit index, or zero if there was an error
155     */
156    public long getCommitIndex() {
157        return this.commitIndex;
158    }
159
160    /**
161     * Get the minimum required leader lease timeout value to commit, if any.
162     *
163     * @return minimum leader lease timeout, or null if there is none
164     */
165    public Timestamp getCommitLeaderLeaseTimeout() {
166        return this.commitLeaderLeaseTimeout;
167    }
168
169    /**
170     * Get the error message in case of failure.
171     *
172     * @return failure message, or null if there was no error
173     */
174    public String getErrorMessage() {
175        return this.errorMessage;
176    }
177
178// Message
179
180    @Override
181    public boolean isLeaderMessage() {
182        return true;
183    }
184
185    @Override
186    public void visit(MessageSwitch handler) {
187        handler.caseCommitResponse(this);
188    }
189
190    @Override
191    public void writeTo(ByteBuffer dest) {
192        super.writeTo(dest);
193        LongEncoder.write(dest, this.txId);
194        LongEncoder.write(dest, this.commitTerm);
195        LongEncoder.write(dest, this.commitIndex);
196        Message.putBoolean(dest, this.commitLeaderLeaseTimeout != null);
197        if (this.commitLeaderLeaseTimeout != null)
198            Message.putTimestamp(dest, this.commitLeaderLeaseTimeout);
199        Message.putBoolean(dest, this.errorMessage != null);
200        if (this.errorMessage != null)
201            Message.putString(dest, this.errorMessage);
202    }
203
204    @Override
205    protected int calculateSize() {
206        return super.calculateSize()
207          + LongEncoder.encodeLength(this.txId)
208          + LongEncoder.encodeLength(this.commitTerm)
209          + LongEncoder.encodeLength(this.commitIndex)
210          + 1
211          + (this.commitLeaderLeaseTimeout != null ? Message.calculateSize(this.commitLeaderLeaseTimeout) : 0)
212          + 1
213          + (this.errorMessage != null ? Message.calculateSize(this.errorMessage) : 0);
214    }
215
216// Object
217
218    @Override
219    public String toString() {
220        final boolean success = this.errorMessage == null;
221        return this.getClass().getSimpleName()
222          + "[\"" + this.getSenderId() + "\"->\"" + this.getRecipientId() + "\""
223          + ",clusterId=" + String.format("%08x", this.getClusterId())
224          + ",term=" + this.getTerm()
225          + ",txId=" + this.txId
226          + (success ?
227            ",commit=" + this.commitTerm + "/" + this.commitIndex
228             + (this.commitLeaderLeaseTimeout != null ? "@" + this.commitLeaderLeaseTimeout : "") :
229            ",error=\"" + this.errorMessage + "\"")
230          + "]";
231    }
232}
