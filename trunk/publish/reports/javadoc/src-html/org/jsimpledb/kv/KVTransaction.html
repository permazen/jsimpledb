008package org.jsimpledb.kv;
009
010/**
011 * {@link KVDatabase} transaction API.
012 *
013 * <p>
014 * Provides a transactional view into a {@link KVStore}.
015 * </p>
016 *
017 * <p>
018 * Instances may throw {@link KVTransactionException} during any operation if the transaction cannot be continued.
019 * In particular, {@link StaleTransactionException} is thrown by a transaction that is no longer open, and
020 * {@link RetryTransactionException} is thrown when the transaction should be retried due to a transient
021 * problem (such as a write conflict with another transaction). When {@link RetryTransactionException} is thrown by
022 * {@link #commit}, the transaction may have actually been committed. Therefore, transactions should be written to be idempotent.
023 * When any {@link KVTransactionException} is thrown, the transaction must still support invoking {@link #rollback},
024 * but all other operations may throw {@link StaleTransactionException}.
025 * </p>
026 *
027 * <p>
028 * If an instance throws a {@link KVTransactionException}, the transaction should be implicitly rolled back.
029 * </p>
030 *
031 * <p>
032 * Implementations must throw {@link StaleTransactionException} if {@link #commit} or {@link #rollback} has already
033 * been invoked, or if the {@link KVTransaction} instance is no longer usable for some other reason. In particular,
034 * implementations should throw {@link TransactionTimeoutException} if an operation is attempted on a transaction
035 * that has been held open past some maximum allowed time limit.
036 * </p>
037 *
038 * <p>
039 * Implementations are responsible for ensuring modifications to {@code byte[]} arrays after method
040 * invocations do no harm. This usually means {@code byte[]} array parameters and return values must be copied.
041 * </p>
042 *
043 * <p>
044 * Accessing keys that start with {@code 0xff} is not supported and will result in {@link IllegalArgumentException}.
045 * </p>
046 */
047public interface KVTransaction extends KVStore {
048
049    /**
050     * Get the {@link KVDatabase} with which this instance is associated.
051     *
052     * @return associated database
053     */
054    KVDatabase getKVDatabase();
055
056    /**
057     * Change the timeout for this transaction from its default value (optional operation).
058     *
059     * @param timeout transaction timeout in milliseconds, or zero for unlimited
060     * @throws UnsupportedOperationException if this transaction does not support timeouts
061     * @throws IllegalArgumentException if {@code timeout} is negative
062     * @throws StaleTransactionException if this transaction is no longer usable
063     */
064    void setTimeout(long timeout);
065
066    /**
067     * Commit this transaction.
068     *
069     * <p>
070     * Note that if this method throws a {@link RetryTransactionException},
071     * the transaction was either successfully committed or rolled back. In either case,
072     * this instance is no longer usable.
073     * </p>
074     *
075     * @throws StaleTransactionException if this transaction is no longer usable
076     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
077     */
078    void commit();
079
080    /**
081     * Cancel this transaction.
082     *
083     * <p>
084     * After this method returns, this instance is no longer usable.
085     *
086     * <p>
087     * This method may be invoked at any time, even after a previous invocation of
088     * {@link #commit} or {@link #rollback}, in which case the invocation will be ignored.
089     */
090    void rollback();
091}
