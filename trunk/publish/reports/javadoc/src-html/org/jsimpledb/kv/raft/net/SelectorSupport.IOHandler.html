008package org.jsimpledb.kv.raft.net;
009
010import com.google.common.base.Preconditions;
011
012import java.io.IOException;
013import java.nio.channels.ClosedChannelException;
014import java.nio.channels.SelectableChannel;
015import java.nio.channels.SelectionKey;
016import java.nio.channels.Selector;
017import java.util.ArrayList;
018import java.util.Iterator;
019
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022
023/**
024 * Support superclass for classes that perform NIO selection operations.
025 */
026public abstract class SelectorSupport {
027
028    protected final Logger log = LoggerFactory.getLogger(this.getClass());
029
030    /**
031     * Create a new {@link SelectionKey} by registering the given channel on the given {@link Selector}
032     * and associating the specified {@link IOHandler} to handle I/O ready conditions.
033     *
034     * <p>
035     * Initially, no I/O operations will be selected. Use {@link #selectFor selectFor()} to add/remove them.
036     *
037     * @param selector NIO selector
038     * @param channel I/O channel
039     * @param handler I/O handler
040     * @return new selection key
041     * @throws IllegalArgumentException if any parameter is null
042     * @throws ClosedChannelException if {@code channel} is closed
043     */
044    protected SelectionKey createSelectionKey(Selector selector, SelectableChannel channel, IOHandler handler)
045      throws ClosedChannelException {
046
047        // Sanity check
048        Preconditions.checkArgument(selector != null, "null selector");
049        Preconditions.checkArgument(channel != null, "null channel");
050        Preconditions.checkArgument(handler != null, "null handler");
051
052        // Register channel with our selector
053        return channel.register(selector, 0, handler);
054    }
055
056    /**
057     * Enable or disable listening for the specified I/O operation(s).
058     *
059     * @param selectionKey selection key; may be null, in which case this method does nothing
060     * @param ops I/O operations
061     * @param enabled true to enable, false to disable
062     */
063    protected void selectFor(SelectionKey selectionKey, int ops, boolean enabled) {
064        if (selectionKey == null)
065            return;
066        final int currentOps = selectionKey.interestOps();
067        selectionKey.interestOps(enabled ? currentOps | ops : currentOps & ~ops);
068    }
069
070    /**
071     * Handle any ready I/O operations available from the given {@link Selector}.
072     *
073     * <p>
074     * This assumes an {@link IOHandler} is assocated with each {@link SelectionKey} (e.g., they were created
075     * via {@link #createSelectionKey createSelectionKey()}).
076     *
077     * @param selector NIO selector
078     */
079    protected void handleIO(Selector selector) {
080        for (Iterator<SelectionKey> i = selector.selectedKeys().iterator(); i.hasNext(); ) {
081            final SelectionKey key = i.next();
082            i.remove();
083            final IOHandler handler = (IOHandler)key.attachment();
084            if (this.log.isTraceEnabled())
085                this.log.trace("I/O ready: key=" + dbg(key) + " handler=" + handler);
086            try {
087                handler.serviceIO(key);
088            } catch (IOException e) {
089                if (this.log.isDebugEnabled())
090                    this.log.debug("I/O error from " + handler, e);
091                handler.close(e);
092            } catch (Throwable t) {
093                this.log.error("service error from " + handler, t);
094                handler.close(t);
095            }
096        }
097    }
098
099    /**
100     * Get a debug description of the given keys.
101     *
102     * @param keys selection keys
103     * @return debug description
104     */
105    protected static String dbg(Iterable<? extends SelectionKey> keys) {
106        final ArrayList<String> strings = new ArrayList<>();
107        for (SelectionKey key : keys)
108            strings.add(dbg(key));
109        return strings.toString();
110    }
111
112    /**
113     * Get a debug description of the given key.
114     *
115     * @param key selection key
116     * @return debug description
117     */
118    protected static String dbg(SelectionKey key) {
119        try {
120            return "Key[interest=" + dbgOps(key.interestOps()) + ",ready="
121              + dbgOps(key.readyOps()) + ",obj=" + key.attachment() + "]";
122        } catch (java.nio.channels.CancelledKeyException e) {
123            return "Key[canceled]";
124        }
125    }
126
127    /**
128     * Get a debug description of the given I/O ops.
129     *
130     * @param ops I/O operation bits
131     * @return debug description
132     */
133    protected static String dbgOps(int ops) {
134        final StringBuilder buf = new StringBuilder(4);
135        if ((ops & SelectionKey.OP_ACCEPT) != 0)
136            buf.append("A");
137        if ((ops & SelectionKey.OP_CONNECT) != 0)
138            buf.append("C");
139        if ((ops & SelectionKey.OP_READ) != 0)
140            buf.append("R");
141        if ((ops & SelectionKey.OP_WRITE) != 0)
142            buf.append("W");
143        return buf.toString();
144    }
145
146// IOHandler
147
148    /**
149     * Callback interface used by {@link SelectorSupport}.
150     */
151    public interface IOHandler {
152
153        /**
154         * Handle ready I/O.
155         *
156         * @param key selection key
157         * @throws IOException if an I/O error occurs; this will result in this instance being {@link #close close()}'ed
158         */
159        void serviceIO(SelectionKey key) throws IOException;
160
161        /**
162         * Notification that an exception was thrown by {@link #serviceIO serviceIO()}.
163         * Typically this method will close the associated channel, which implicitly unregisters
164         * the associated {@link SelectionKey}s.
165         *
166         * <p>
167         * This method must be idempotent.
168         *
169         * @param cause the error that occurred
170         */
171        void close(Throwable cause);
172    }
173}
