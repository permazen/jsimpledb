008package org.jsimpledb.kv.mvcc;
009
010import java.util.Collections;
011import java.util.HashSet;
012import java.util.Set;
013
014import org.jsimpledb.kv.KVStore;
015
016/**
017 * Represents one {@link SnapshotKVDatabase} MVCC version.
018 *
019 * <p>
020 * Information included:
021 *  <ul>
022 *  <li>MVCC version number</li>
023 *  <li>{@link KVStore} snapshot</li>
024 *  <li>Open transactions based on this version</li>
025 *  <li>The {@link Writes} of the transaction that was committed on this version, if any</li>
026 *  </ul>
027 */
028public class SnapshotVersion {
029
030    private final long version;
031    private final KVStore snapshot;
032    private final HashSet<SnapshotKVTransaction> openTransactions = new HashSet<>(2);
033
034    private Writes committedWrites;
035
036    /**
037     * Constructor.
038     *
039     * @param version version number
040     * @param snapshot database snapshot
041     */
042    SnapshotVersion(long version, KVStore snapshot) {
043        this.version = version;
044        this.snapshot = snapshot;
045    }
046
047    /**
048     * Get this instance's unique version number.
049     *
050     * @return MVCC version number
051     */
052    public long getVersion() {
053        return this.version;
054    }
055
056    /**
057     * Get the {@link KVStore} view of this version's snapshot.
058     *
059     * @return unmodifiable {@link KVStore}
060     */
061    public KVStore getSnapshot() {
062        return this.snapshot;
063    }
064
065    /**
066     * Get transactions based on this version's snapshot that are still open.
067     *
068     * @return immutable view of open transactions associated with this version
069     */
070    public Set<SnapshotKVTransaction> getOpenTransactions() {
071        return Collections.unmodifiableSet(this.openTransactions);
072    }
073
074    /**
075     * Get the {@link Writes} of the transaction based on this version's snapshot that was eventually committed, if any.
076     *
077     * @return writes of the transaction committed on this version, or null if there is none
078     */
079    public Writes getCommittedWrites() {
080        return this.committedWrites;
081    }
082
083// Package methods
084
085    void addOpenTransaction(SnapshotKVTransaction tx) {
086        this.openTransactions.add(tx);
087    }
088
089    void removeOpenTransaction(SnapshotKVTransaction tx) {
090        this.openTransactions.remove(tx);
091    }
092
093    void setCommittedWrites(Writes committedWrites) {
094        this.committedWrites = committedWrites;
095    }
096
097// Object
098
099    @Override
100    public String toString() {
101        return this.getClass().getSimpleName()
102          + "[vers=" + this.version
103          + ",snapshot=" + this.snapshot
104          + (!this.openTransactions.isEmpty() ? ",openTx=" + this.openTransactions : "")
105          + (this.committedWrites != null ? ",writes=" + this.committedWrites : "")
106          + "]";
107    }
108}
