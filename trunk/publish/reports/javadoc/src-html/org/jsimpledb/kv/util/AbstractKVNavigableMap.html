030/**
031 * {@link java.util.NavigableMap} support superclass for maps backed by keys and values encoded as {@code byte[]}
032 * arrays in a {@link KVStore}, and whose key sort order is consistent with the {@code byte[]} key encoding.
033 * There must be an equivalence between map keys and {@code byte[]} key encodings (i.e., there must be
034 * only one valid encoding per map key).
035 *
036 * <p><b>Subclass Methods</b></p>
037 *
038 * <p>
039 * Subclasses must implement the {@linkplain #encodeKey encodeKey()}, {@linkplain #decodeKey decodeKey()},
040 * and {@linkplain #decodeValue decodeValue()} methods to convert keys and value to/from {@link KVStore} keys and values,
041 * and {@link #createSubMap(boolean, KeyRange, KeyFilter, Bounds) createSubMap()}
042 * to allow creating reversed and restricted range sub-maps.
043 *
044 * <p>
045 * Subclasses must also implement {@link #comparator comparator()}, and the resulting sort order must be consistent with
046 * the sort order of the encoded {@code byte[]} keys (possibly {@link #reversed}).
047 * </p>
048 *
049 * <p>
050 * This class provides a read-only implementation; for a mutable implementation, subclasses should also implement
051 * {@link #put put()}, {@link #remove remove()}, and {@link #clear}; note, these methods must verify the key
052 * {@link #isVisible isVisible()} before making any changes.
053 * </p>
054 *
055 * <p>
056 * Additional subclass notes:
057 * <ul>
058 *  <li>{@link #navigableKeySet} returns a {@link Set} for which:
059 *  <ul>
060 *      <li>The {@link Set#add add()} method is not supported.</li>
061 *      <li>The {@link Set#remove remove()} method and the {@link Iterator#remove remove()} method of the
062 *          associated {@link Iterator} delegate to this instance's {@link #remove remove()} method.</li>
063 *      <li>The {@link Set#clear clear()} method delegates to this instance's {@link #clear} method.</li>
064 *  </ul>
065 *  <li>{@link #entrySet} returns a {@link Set} for which:
066 *  <ul>
067 *      <li>The {@link Set#add add()} method is not supported.</li>
068 *      <li>The {@link Set#remove remove()} method and the {@link Iterator#remove remove()} method of the
069 *          associated {@link Iterator} delegate to this instance's {@link #remove remove()} method (but
070 *          only when the {@link java.util.Map.Entry} is still contained in the map).
071 *      <li>The {@link java.util.Map.Entry} elements' {@link java.util.Map.Entry#setValue setValue()} method delegates to
072 *          this instance's {@link #put put()} method (but only when the {@link java.util.Map.Entry} is still contained in the map).
073 *      <li>The {@link Set#clear clear()} method delegates to this instance's {@link #clear} method.</li>
074 *  </ul>
075 * </ul>
076 *
077 * <p><b>Prefix Mode</b></p>
078 *
079 * <p>
080 * Instances support "prefix mode" where the {@code byte[]} keys may have arbitrary trailing garbage, which is ignored,
081 * and so by definition no key can be a prefix of any other key. The length of the prefix is determined implicitly by the
082 * number of bytes produced by {@link #encodeKey encodeKey()} or consumed by {@link #decodeKey decodeKey()}.
083 * When not in prefix mode, {@link #decodeKey decodeKey()} <b>must</b> consume the entire key to preserve correct semantics.
084 * </p>
085 *
086 * <p><b>Key Restrictions</b></p>
087 *
088 * <p>
089 * Instances are configured with an (optional) {@link KeyRange}; when {@linkplain #bounds range restriction} is in
090 * effect, this key range corresponds to the bounds.
091 * </p>
092 *
093 * <p>
094 * Instances also support filtering visible keys using a {@link KeyFilter}; see {@link #filterKeys filterKeys()}.
095 * To be {@linkplain #isVisible} in the map, keys must both be in the {@link KeyRange} and pass the {@link KeyFilter}.
096 * </p>
097 *
098 * <p><b>Concurrent Modifications</b></p>
099 *
100 * <p>
101 * This implementation never throws {@link java.util.ConcurrentModificationException}; instead, iterators always
102 * see the most up-to-date state of the associated {@link KVStore}.
103 * </p>
104 *
105 * @see AbstractKVNavigableSet
106 * @param <K> key type
107 * @param <V> value type
108 */
109@SuppressWarnings("serial")
110public abstract class AbstractKVNavigableMap<K, V> extends AbstractNavigableMap<K, V> {
111
112    /**
113     * The underlying {@link KVStore}.
114     */
115    protected final KVStore kv;
116
117    /**
118     * Whether we are in "prefix" mode.

121
122    /**
123     * Whether the ordering of this instance is reversed.
124     */

127    /**
128     * Key range, or null for the entire range.

137// Constructors
138
139    /**
140     * Convenience constructor for when there are no range restrictions.
141     *
142     * @param kv underlying {@link KVStore}
143     * @param prefixMode whether to allow keys to have trailing garbage
144     * @throws IllegalArgumentException if {@code kv} is null
145     */
146    protected AbstractKVNavigableMap(KVStore kv, boolean prefixMode) {
147        this(kv, prefixMode, (KeyRange)null);
148    }

150    /**
151     * Convenience constructor for when the range of visible {@link KVStore} keys is all keys sharing a given {@code byte[]} prefix.
152     *
153     * @param kv underlying {@link KVStore}
154     * @param prefixMode whether to allow keys to have trailing garbage
155     * @param prefix prefix defining minimum and maximum keys
156     * @throws IllegalArgumentException if {@code kv} is null
157     * @throws IllegalArgumentException if {@code prefix} is null or empty
158     */
159    protected AbstractKVNavigableMap(KVStore kv, boolean prefixMode, byte[] prefix) {
160        this(kv, prefixMode, KeyRange.forPrefix(prefix));
161    }
162
163    /**
164     * Primary constructor.
165     *
166     * @param kv underlying {@link KVStore}
167     * @param prefixMode whether to allow keys to have trailing garbage
168     * @param keyRange key range restriction, or null for none
169     * @throws IllegalArgumentException if {@code kv} is null
170     */
171    protected AbstractKVNavigableMap(KVStore kv, boolean prefixMode, KeyRange keyRange) {
172        this(kv, prefixMode, false, keyRange, null, new Bounds<K>());
173    }
174
175    /**
176     * Internal constructor. Used for creating sub-maps and reversed views.
177     *
178     * <p>
179     * Note: if {@code bounds} are set, then {@code keyRange} must exclude all keys outside of those bounds.
180     * </p>
181     *
182     * @param kv underlying {@link KVStore}
183     * @param prefixMode whether to allow keys to have trailing garbage
184     * @param reversed whether ordering is reversed (implies {@code bounds} are also inverted, but <i>not</i> {@code keyRange})
185     * @param keyRange key range restriction, or null for none
186     * @param keyFilter key filter, or null for none
187     * @param bounds range restriction
188     * @throws IllegalArgumentException if {@code kv} or {@code bounds} is null
189     */
190    protected AbstractKVNavigableMap(KVStore kv, boolean prefixMode, boolean reversed,
191      KeyRange keyRange, KeyFilter keyFilter, Bounds<K> bounds) {
192        super(bounds);
193        if (kv == null)
194            throw new IllegalArgumentException("null kv");
195        this.kv = kv;
196        this.prefixMode = prefixMode;
197        this.reversed = reversed;
198        this.keyRange = keyRange;
199        this.keyFilter = keyFilter;
200    }
201
202    @Override
203    public V get(Object obj) {
204
205        // Encode key and check visibility
206        final byte[] key = this.encodeVisibleKey(obj, false);
207        if (key == null)
208            return null;
209
210        // Find key, or some longer key with the same prefix in prefix mode
211        final KVPair pair;
212        if (this.prefixMode) {
213            pair = this.kv.getAtLeast(key);
214            if (pair == null || !ByteUtil.isPrefixOf(key, pair.getKey()))
215                return null;
216        } else {
217            final byte[] value = this.kv.get(key);
218            if (value == null)
219                return null;
220            pair = new KVPair(key, value);
221        }
222
223        // Decode value
224        return this.decodeValue(pair);
225    }
226
227    @Override
228    public Set<Map.Entry<K, V>> entrySet() {
229        return this.new EntrySet();
230    }
231
232    @Override
233    public NavigableSet<K> navigableKeySet() {
234        return new KeySet();
235    }

237    /**
238     * Create a view of this instance with additional filtering applied to the underlying {@code byte[]} keys.
239     * Any map entry for which the corresponding key does not pass {@code keyFilter} will be effectively hidden from view.
240     *
241     * <p>
242     * The restrictions of the given {@link KeyFilter} will be added to any current {@link KeyFilter} restrictions on this instance.
243     * The {@link #bounds} associated with this instance will not change.
244     * </p>
245     *
246     * @param keyFilter additional key filtering to apply
247     * @return filtered view of this instance
248     * @throws IllegalArgumentException if {@code keyFilter} is null
249     */
250    public NavigableMap<K, V> filterKeys(KeyFilter keyFilter) {
251        if (keyFilter == null)
252            throw new IllegalArgumentException("null keyFilter");
253        if (this.keyFilter != null)
254            keyFilter = KeyFilterUtil.intersection(keyFilter, this.keyFilter);
255        return this.createSubMap(this.reversed, this.keyRange, keyFilter, this.bounds);
256    }
257
258    @Override
259    protected boolean isWithinLowerBound(K key) {
260        if (!super.isWithinLowerBound(key))
261            return false;
262        if (this.keyRange == null)
263            return true;
264        final ByteWriter writer = new ByteWriter();
265        this.encodeKey(writer, key);
266        return KeyRange.compare(writer.getBytes(), this.keyRange.getMin()) >= 0;
267    }
268
269    @Override
270    protected boolean isWithinUpperBound(K key) {
271        if (!super.isWithinUpperBound(key))
272            return false;
273        if (this.keyRange == null)
274            return true;
275        final ByteWriter writer = new ByteWriter();
276        this.encodeKey(writer, key);
277        return KeyRange.compare(writer.getBytes(), this.keyRange.getMax()) < 0;
278    }
279
280    @Override
281    protected final NavigableMap<K, V> createSubMap(boolean reverse, Bounds<K> newBounds) {
282
283        // Determine the direction of the new sub-map
284        final boolean newReversed = this.reversed ^ reverse;
285
286        // Determine new min and max keys
287        final KeyRange newKeyRange = this.buildKeyRange(newReversed ? newBounds.reverse() : newBounds);
288
289        // Create submap
290        return this.createSubMap(newReversed, newKeyRange, this.keyFilter, newBounds);
291    }
292
293    /**
294     * Create a (possibly reversed) view of this instance with (possibly) tighter lower and/or upper bounds and
295     * the given {@link KeyFilter}, if any.
296     * The bounds are consistent with the reversed ordering (i.e., reversed if {@code reverse} is true)
297     * and have already been range-checked against this instance's bounds.
298     *
299     * @param newReversed whether the new map's ordering should be reversed (implies {@code newBounds} are also inverted,
300     *  but <i>not</i> {@code keyRange}); note: means "absolutely" reversed, not relative to this instance
301     * @param newKeyRange new key range, or null for none; will be consistent with {@code bounds}, if any
302     * @param newKeyFilter new key filter, or null for none
303     * @param newBounds new bounds
304     * @return restricted and/or filtered view of this instance
305     * @throws IllegalArgumentException if {@code newBounds} is null
306     */
307    protected abstract NavigableMap<K, V> createSubMap(boolean newReversed,
308      KeyRange newKeyRange, KeyFilter newKeyFilter, Bounds<K> newBounds);
309
310    /**
311     * Encode the given key object into a {@code byte[]} key.
312     * Note that this method must throw {@link IllegalArgumentException}, not {@link ClassCastException}
313     * or {@code NullPointerException}, if {@code obj} does not have the correct type or is an illegal null value.
314     *
315     * @param writer output for encoded {@code byte[]} key corresponding to {@code obj}
316     * @param obj map key object
317     * @throws IllegalArgumentException if {@code obj} is not of the required Java type supported by this set
318     * @throws IllegalArgumentException if {@code obj} is null and this set does not support null elements
319     */
320    protected abstract void encodeKey(ByteWriter writer, Object obj);
321
322    /**
323     * Decode a key object from an encoded {@code byte[]} key.
324     *
325     * <p>
326     * If not in prefix mode, all of {@code reader} must be consumed; otherwise, the consumed portion
327     * is the prefix and any following keys with the same prefix are ignored.
328     * </p>
329     *
330     * @param reader input for encoded bytes
331     * @return decoded map key
332     */
333    protected abstract K decodeKey(ByteReader reader);
334
335    /**
336     * Decode a value object from an encoded {@code byte[]} key/value pair.
337     *
338     * @param pair key/value pair
339     * @return decoded map value
340     */
341    protected abstract V decodeValue(KVPair pair);
342
343    /**
344     * Determine if the given {@code byte[]} key is visible in this map according to the configured
345     * {@link KeyRange} and/or {@link KeyFilter}, if any.
346     *
347     * @param key key to test
348     * @return true if key is visible
349     * @throws IllegalArgumentException if {@code key} is null
350     * @see #filterKeys filterKeys()
351     */
352    protected boolean isVisible(byte[] key) {
353        return (this.keyRange == null || this.keyRange.contains(key))
354          && (this.keyFilter == null || this.keyFilter.contains(key));
355    }
356
357    /**
358     * Encode the given key object, if possible, and verify the corresponding {@code byte[]} key is visible,
359     * otherwise return null or throw an exception.
360     * Delegates to {@link #encodeKey(ByteWriter, Object)} to attempt the actual encoding.
361     *
362     * @param obj key object to encode, possibly null
363     * @param fail whether, if {@code obj} can't be encoded, to throw an exception (true) or return null (false)
364     * @return encoed key for {@code obj}, or null if {@code fail} is false and {@code obj} has the wrong type or is out of bounds
365     * @throws IllegalArgumentException if {@code fail} is true and {@code obj} has the wrong type
366     * @throws IllegalArgumentException if {@code fail} is true and the resulting key is not {@linkplain #isVisible visible}
367     */
368    protected byte[] encodeVisibleKey(Object obj, boolean fail) {
369        final ByteWriter writer = new ByteWriter();
370        try {
371            this.encodeKey(writer, obj);
372        } catch (IllegalArgumentException e) {
373            if (!fail)
374                return null;
375            throw e;
376        }
377        final byte[] key = writer.getBytes();
378        if (this.keyRange != null && !this.keyRange.contains(key)) {
379            if (fail)
380                throw new IllegalArgumentException("key is out of bounds: " + obj);
381            return null;
382        }
383        if (this.keyFilter != null && !this.keyFilter.contains(key)) {
384            if (fail)
385                throw new IllegalArgumentException("key is filtered out: " + obj);
386            return null;
387        }
388        return key;
389    }
390
391    /**
392     * Derive a new {@link KeyRange} from (possibly) new element bounds. The given bounds must <i>not</i> ever be reversed.
393     */
394    private KeyRange buildKeyRange(Bounds<K> bounds) {
395        final byte[] minKey = this.keyRange != null ? this.keyRange.getMin() : null;
396        final byte[] maxKey = this.keyRange != null ? this.keyRange.getMax() : null;
397        byte[] newMinKey;
398        byte[] newMaxKey;
399        switch (bounds.getLowerBoundType()) {
400        case NONE:
401            newMinKey = minKey;
402            break;
403        default:
404            final ByteWriter writer = new ByteWriter();
405            this.encodeKey(writer, bounds.getLowerBound());
406            newMinKey = writer.getBytes();
407            if (!bounds.getLowerBoundType().isInclusive())
408                newMinKey = this.prefixMode ? ByteUtil.getKeyAfterPrefix(newMinKey) : ByteUtil.getNextKey(newMinKey);
409            if (minKey != null)
410                newMinKey = ByteUtil.max(newMinKey, minKey);
411            if (maxKey != null)
412                newMinKey = ByteUtil.min(newMinKey, maxKey);
413            break;
414        }
415        switch (bounds.getUpperBoundType()) {
416        case NONE:
417            newMaxKey = maxKey;
418            break;
419        default:
420            final ByteWriter writer = new ByteWriter();
421            this.encodeKey(writer, bounds.getUpperBound());
422            newMaxKey = writer.getBytes();
423            if (bounds.getUpperBoundType().isInclusive())
424                newMaxKey = this.prefixMode ? ByteUtil.getKeyAfterPrefix(newMaxKey) : ByteUtil.getNextKey(newMaxKey);
425            if (maxKey != null)
426                newMaxKey = ByteUtil.min(newMaxKey, maxKey);
427            if (minKey != null)
428                newMaxKey = ByteUtil.max(newMaxKey, minKey);
429            break;
430        }
431        return new KeyRange(newMinKey != null ? newMinKey : ByteUtil.EMPTY, newMaxKey);
432    }

437
438        KeySet() {
439            super(AbstractKVNavigableMap.this.kv, AbstractKVNavigableMap.this.prefixMode, AbstractKVNavigableMap.this.reversed,

479    private class EntrySet extends AbstractIterationSet<Map.Entry<K, V>> {
480
481        @Override
482        public Iterator<Map.Entry<K, V>> iterator() {
483            return new AbstractKVIterator<Map.Entry<K, V>>(AbstractKVNavigableMap.this.kv, AbstractKVNavigableMap.this.prefixMode,
484              AbstractKVNavigableMap.this.reversed, AbstractKVNavigableMap.this.keyRange, AbstractKVNavigableMap.this.keyFilter) {
485
486                @Override
487                protected Map.Entry<K, V> decodePair(KVPair pair, ByteReader keyReader) {
488                    final K key = AbstractKVNavigableMap.this.decodeKey(keyReader);
489                    final V value = AbstractKVNavigableMap.this.decodeValue(pair);
490                    return new MapEntry(key, value);
491                }
492
493                @Override
494                protected void doRemove(Map.Entry<K, V> entry, KVPair pair) {
495                    AbstractKVNavigableMap.this.entrySet().remove(entry);
496                }
497            };
498        }
499
500        @Override
501        @SuppressWarnings("unchecked")
502        public boolean contains(Object obj) {
503
504            // Check type
505            if (!(obj instanceof Map.Entry))
506                return false;
507            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
508
509            // Find key
510            if (!AbstractKVNavigableMap.this.containsKey(entry.getKey()))
511                return false;
512
513            // Compare key/value pair
514            final K key = (K)entry.getKey();
515            final V value = AbstractKVNavigableMap.this.get(entry.getKey());
516            return new MapEntry(key, value).equals(entry);
517        }
518
519        @Override
520        @SuppressWarnings("unchecked")
521        public boolean remove(Object obj) {
522
523            // Check type
524            if (!(obj instanceof Map.Entry))
525                return false;
526            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
527
528            // Find key
529            if (!AbstractKVNavigableMap.this.containsKey(entry.getKey()))
530                return false;
531
532            // Compare key/value pair and remove entry (if contained)
533            final K key = (K)entry.getKey();
534            final V value = AbstractKVNavigableMap.this.get(entry.getKey());
535            if (new MapEntry(key, value).equals(entry)) {
536                AbstractKVNavigableMap.this.remove(key);
537                return true;
538            }
539
540            // Not found
541            return false;
542        }
543
544        @Override
545        public void clear() {
546            AbstractKVNavigableMap.this.clear();
547        }
548    }
