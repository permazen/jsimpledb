008package org.jsimpledb.kv.raft.msg;
009
010import com.google.common.base.Preconditions;
011
012import java.nio.ByteBuffer;
013
014import org.jsimpledb.kv.raft.Timestamp;
015import org.jsimpledb.util.LongEncoder;
016
017/**
018 * Sent from leaders to followers to probe the follower's log state and/or append an entry to their log.
019 *
020 * <p>
021 * Instances also provide the {@linkplain #getLeaderLeaseTimeout leader's lease timeout} value, which is
022 * used to commit read-only transactions, as well as a {@linkplain #getLeaderTimestamp leader timestamp}
023 * which should be reflected back in the corresponding {@link AppendResponse}.
024 */
025public class AppendRequest extends Message {
026
027    private final Timestamp leaderTimestamp;        // leader's timestamp for this request
028    private final Timestamp leaderLeaseTimeout;     // earliest leader timestamp at which time leader could be deposed
029    private final long leaderCommit;                // index of highest log entry known to be committed
030    private final long prevLogTerm;                 // term of previous log entry
031    private final long prevLogIndex;                // index of previous log entry
032    private final long logEntryTerm;                // term corresponding to log entry, or zero if this is a "probe"
033
034    private ByteBuffer mutationData;                // serialized mutations, if not a probe and not from follower transaction
035    private boolean mutationDataInvalid;            // mutationData has already been grabbed
036
037// Constructors
038
039    /**
040     * Construtor for a "probe" that does not contain a log entry.
041     *
042     * @param clusterId cluster ID
043     * @param senderId identity of sender
044     * @param recipientId identity of recipient
045     * @param term sender's current term
046     * @param leaderTimestamp leader's timestamp for this request
047     * @param leaderLeaseTimeout earliest leader timestamp at which leader could be deposed
048     * @param leaderCommit current commit index for sender
049     * @param prevLogTerm term of the log entry just prior to this one
050     * @param prevLogIndex index of the log entry just prior to this one
051     */
052    public AppendRequest(int clusterId, String senderId, String recipientId, long term,
053      Timestamp leaderTimestamp, Timestamp leaderLeaseTimeout, long leaderCommit, long prevLogTerm, long prevLogIndex) {
054        this(clusterId, senderId, recipientId, term, leaderTimestamp, leaderLeaseTimeout, leaderCommit,
055          prevLogTerm, prevLogIndex, 0, null);
056    }
057
058    /**
059     * Constructor for a request that contains an actual log entry.
060     *
061     * @param clusterId cluster ID
062     * @param senderId identity of sender
063     * @param recipientId identity of recipient
064     * @param term sender's current term
065     * @param leaderTimestamp leader's timestamp for this request
066     * @param leaderLeaseTimeout earliest leader timestamp at which leader could be deposed
067     * @param leaderCommit current commit index for sender
068     * @param prevLogTerm term of the log entry just prior to this one
069     * @param prevLogIndex index of the log entry just prior to this one
070     * @param logEntryTerm term of this log entry
071     * @param mutationData log entry serialized mutations, or null if follower should have the data already
072     */
073    public AppendRequest(int clusterId, String senderId, String recipientId, long term, Timestamp leaderTimestamp,
074      Timestamp leaderLeaseTimeout, long leaderCommit, long prevLogTerm, long prevLogIndex, long logEntryTerm,
075      ByteBuffer mutationData) {
076        super(Message.APPEND_REQUEST_TYPE, clusterId, senderId, recipientId, term);
077        this.leaderTimestamp = leaderTimestamp;
078        this.leaderLeaseTimeout = leaderLeaseTimeout;
079        this.leaderCommit = leaderCommit;
080        this.prevLogTerm = prevLogTerm;
081        this.prevLogIndex = prevLogIndex;
082        this.logEntryTerm = logEntryTerm;
083        this.mutationData = mutationData;
084        this.checkArguments();
085    }
086
087    AppendRequest(ByteBuffer buf) {
088        super(Message.APPEND_REQUEST_TYPE, buf);
089        this.leaderTimestamp = Message.getTimestamp(buf);
090        this.leaderLeaseTimeout = this.leaderTimestamp.offset((int)LongEncoder.read(buf));
091        this.leaderCommit = LongEncoder.read(buf);
092        this.prevLogTerm = LongEncoder.read(buf);
093        this.prevLogIndex = LongEncoder.read(buf);
094        this.logEntryTerm = LongEncoder.read(buf);
095        this.mutationData = this.logEntryTerm != 0 && Message.getBoolean(buf) ? Message.getByteBuffer(buf) : null;
096        this.checkArguments();
097    }
098
099    @Override
100    void checkArguments() {
101        super.checkArguments();
102        Preconditions.checkArgument(this.leaderTimestamp != null);
103        Preconditions.checkArgument(this.leaderLeaseTimeout != null);
104        Preconditions.checkArgument(this.leaderCommit >= 0);
105        Preconditions.checkArgument(this.prevLogTerm >= 0);
106        Preconditions.checkArgument(this.prevLogIndex >= 0);
107        Preconditions.checkArgument(this.logEntryTerm >= 0);
108        Preconditions.checkArgument(this.mutationData == null || this.logEntryTerm > 0);
109    }
110
111// Properties
112
113    public Timestamp getLeaderTimestamp() {
114        return this.leaderTimestamp;
115    }
116
117    public Timestamp getLeaderLeaseTimeout() {
118        return this.leaderLeaseTimeout;
119    }
120
121    public long getLeaderCommit() {
122        return this.leaderCommit;
123    }
124
125    public long getPrevLogTerm() {
126        return this.prevLogTerm;
127    }
128
129    public long getPrevLogIndex() {
130        return this.prevLogIndex;
131    }
132
133    public boolean isProbe() {
134        return this.logEntryTerm == 0;
135    }
136
137    public long getLogEntryTerm() {
138        return this.logEntryTerm;
139    }
140
141    /**
142     * Get the serialized data for the log entry, if any.
143     * Returns null if this is a probe or follower is expected to already have the data from a transaction.
144     *
145     * <p>
146     * This method may only be invoked once.
147     *
148     * @return log entry serialized mutations, or null if this message does not contain data
149     * @throws IllegalStateException if this method has already been invoked
150     */
151    public ByteBuffer getMutationData() {
152        Preconditions.checkState(!this.mutationDataInvalid);
153        final ByteBuffer result = this.mutationData;
154        this.mutationData = null;
155        this.mutationDataInvalid = true;
156        return result;
157    }
158
159// Message
160
161    @Override
162    public boolean isLeaderMessage() {
163        return true;
164    }
165
166    @Override
167    public void visit(MessageSwitch handler) {
168        handler.caseAppendRequest(this);
169    }
170
171    @Override
172    public void writeTo(ByteBuffer dest) {
173        Preconditions.checkState(!this.mutationDataInvalid);
174        super.writeTo(dest);
175        Message.putTimestamp(dest, this.leaderTimestamp);
176        LongEncoder.write(dest, this.leaderLeaseTimeout.offsetFrom(this.leaderTimestamp));
177        LongEncoder.write(dest, this.leaderCommit);
178        LongEncoder.write(dest, this.prevLogTerm);
179        LongEncoder.write(dest, this.prevLogIndex);
180        LongEncoder.write(dest, this.logEntryTerm);
181        if (this.logEntryTerm != 0) {
182            Message.putBoolean(dest, this.mutationData != null);
183            if (this.mutationData != null)
184                Message.putByteBuffer(dest, this.mutationData);
185        }
186    }
187
188    @Override
189    protected int calculateSize() {
190        Preconditions.checkState(!this.mutationDataInvalid);
191        return super.calculateSize()
192          + Message.calculateSize(this.leaderTimestamp)
193          + LongEncoder.encodeLength(this.leaderLeaseTimeout.offsetFrom(this.leaderTimestamp))
194          + LongEncoder.encodeLength(this.leaderCommit)
195          + LongEncoder.encodeLength(this.prevLogTerm)
196          + LongEncoder.encodeLength(this.prevLogIndex)
197          + LongEncoder.encodeLength(this.logEntryTerm)
198          + (this.logEntryTerm != 0 ? 1 + (this.mutationData != null ? Message.calculateSize(this.mutationData) : 0) : 0);
199    }
200
201// Object
202
203    @Override
204    public String toString() {
205        return this.getClass().getSimpleName()
206          + "[\"" + this.getSenderId() + "\"->\"" + this.getRecipientId() + "\""
207          + ",clusterId=" + String.format("%08x", this.getClusterId())
208          + ",term=" + this.getTerm()
209          + ",leaderTimestamp=" + this.leaderTimestamp
210          + ",leaderLeaseTimeout=" + String.format("%+dms", this.leaderLeaseTimeout.offsetFrom(this.leaderTimestamp))
211          + ",leaderCommit=" + this.leaderCommit
212          + ",prevLog=" + this.prevLogTerm + "/" + this.prevLogIndex
213          + (this.logEntryTerm != 0 ? ",logEntryTerm=" + this.logEntryTerm : "")
214          + (this.mutationData != null ?
215            ",mutationData=" + this.describe(this.mutationData) : this.mutationDataInvalid ? ",mutationData=invalid" : "")
216          + "]";
217    }
218}
