031/**
032 * {@link org.jsimpledb.kv.KVDatabase} implementation based on LevelDB and providing concurrent transactions
033 * and linearizable ACID semantics.
034 *
035 * <p>
036 * A {@linkplain #setDirectory database directory} is the only required configuration property.
037 * Instances may be stopped and (re)started multiple times.
038 * </p>
039 *
040 * <p>
041 * The LevelDB Java implementation is configurable by specifying the {@link DBFactory} implementation.
042 * </p>
043 *
044 * @see <a href="https://github.com/dain/leveldb">leveldb</a>
045 * @see <a href="https://github.com/fusesource/leveldbjni">leveldbjni</a>
046 */
047public class LevelDBKVDatabase extends SnapshotKVDatabase {
048
049    /**
050     * Class name for the {@link DBFactory} provided by <a href="https://github.com/fusesource/leveldbjni">leveldbjni</a>.
051     */
052    public static final String LEVELDBJNI_CLASS_NAME = "org.fusesource.leveldbjni.JniDBFactory";
053
054    /**
055     * Class name for the {@link DBFactory} provided by <a href="https://github.com/dain/leveldb">leveldb</a>.
056     */
057    public static final String LEVELDB_CLASS_NAME = "org.iq80.leveldb.impl.Iq80DBFactory";
058
059    /**
060     * The name of a system property that can be set to override the default {@link DBFactory} logic.
061     * Set to the name of a class that implements {@link DBFactory} and has a zero-arg constructor.
062     */
063    public static final String DB_FACTORY_PROPERTY = LevelDBKVDatabase.class.getName() + ".db_factory";

106    private static DBFactory getDefaultDBFactory() {
107
108        // Get class names to try
109        final ArrayList<String> classNames = new ArrayList<>(3);
110        final String configuredFactoryClass = System.getProperty(DB_FACTORY_PROPERTY, null);
111        if (configuredFactoryClass != null)
112            classNames.add(configuredFactoryClass);
113        classNames.add(LEVELDBJNI_CLASS_NAME);
114        classNames.add(LEVELDB_CLASS_NAME);
115
116        // Find class
117        final ClassLoader loader = Thread.currentThread().getContextClassLoader();
118        final Logger log = LoggerFactory.getLogger(LevelDBKVDatabase.class);
119        for (String className : classNames) {
120            try {
121                return Class.forName(className, false, loader).asSubclass(DBFactory.class).newInstance();
122            } catch (Exception e) {
123                if (log.isDebugEnabled())
124                    log.debug("can't load factory class `" + className + "': " + e);
125                continue;
126            }
127        }
128        throw new RuntimeException("no " + DBFactory.class.getName() + " implementation found; tried: " + classNames);
129    }

131// Lifecycle
132
133    /**
134     * Start this instance. This method must be called prior to creating any transactions.
135     *
136     * <p>
137     * This method is idempotent.
138     * </p>
139     *
140     * @throws IllegalStateException if this instance is not properly configured
141     */
142    @PostConstruct
143    public synchronized void start() {
144
145        // Already started?
146        if (this.db != null)
147            return;
148        this.log.info("starting " + this);
149
150        // Check configuration
151        if (this.directory == null)
152            throw new IllegalStateException("no directory configured");
153
154        // Open database
155        if (this.log.isDebugEnabled())
156            this.log.debug("opening " + this + " LevelDB database");
157        try {
158            this.db = this.factory.open(this.directory, this.options);
159        } catch (IOException e) {
160            throw new RuntimeException("LevelDB database startup failed", e);
161        }
162
163        // Configure KVStore
164        this.setKVStore(new LevelDBKVStore(this.db, new ReadOptions().verifyChecksums(this.options.verifyChecksums()), null));
165
166        // Add shutdown hook so we don't leak native resources
167        if (this.shutdownHookRegistered.compareAndSet(false, true)) {
168            Runtime.getRuntime().addShutdownHook(new Thread() {
169                public void run() {
170                    LevelDBKVDatabase.this.stop();
171                }
172            });
173        }
174    }
175
176    /**
177     * Stop this instance. Does nothing if not {@linkplain #start started} or already stopped.
178     */
179    @PreDestroy
180    public void stop() {
181
182        // Set stopping flag to prevent new transactions from being created
183        synchronized (this) {
184            if (this.db == null || this.stopping)
185                return;
186            this.log.info("stopping " + this);
187            this.stopping = true;
188        }
189
190        // Close any remaining open transactions
191        this.closeTransactions();
192
193        // Finish up
194        synchronized (this) {
195
196            // Sanity check
197            assert this.db != null;
198
199            // Close KVStore
200            ((LevelDBKVStore)this.getKVStore()).close();
201            this.setKVStore(null);
202
203            // Shut down LevelDB database
204            try {
205                if (this.log.isDebugEnabled())
206                    this.log.info("closing " + this + " LevelDB database");
207                this.db.close();
208            } catch (Throwable e) {
209                this.log.error("caught exception closing database during shutdown (ignoring)", e);
210            }
211
212            // Reset state
213            this.db = null;
214            this.stopping = false;
215        }
216    }

365// KVDatabase
366
367    /**
368     * Create a new transaction.
369     *
370     * @throws IllegalStateException if this instance is not {@linkplain #start started}
371     */
372    @Override
373    public synchronized LevelDBKVTransaction createTransaction() {
374
375        // Sanity check
376        if (this.db == null)
377            throw new IllegalStateException("not started");
378        if (this.stopping)
379            throw new IllegalStateException("stop in progress");
380
381        // OK
382        return (LevelDBKVTransaction)super.createTransaction();
383    }

385// SnapshotKVDatabase

387    @Override
388    protected LevelDBKVTransaction createSnapshotKVTransaction(SnapshotVersion versionInfo) {
389        return new LevelDBKVTransaction(this, versionInfo);
390    }
391
392    @Override
393    protected RuntimeException wrapException(SnapshotKVTransaction tx, RuntimeException e) {
394        if (e instanceof DBException)
395            return new KVTransactionException(tx, "LevelDB error", e);
396        return e;
397    }
