008package org.jsimpledb.kv;
009
010import com.google.common.primitives.Bytes;
011
012import java.util.Arrays;
013import java.util.Comparator;
014
015import org.jsimpledb.util.ByteUtil;
016import org.jsimpledb.util.SizeEstimating;
017import org.jsimpledb.util.SizeEstimator;
018
019/**
020 * Represents a contiguous range of {@code byte[]} keys, when keys are sorted in unsigned lexical order.
021 * Instances are defined by an inclusive lower bound and an exclusive upper bound.
022 * The upper bound may be specified as null to represent no maximum.
023 *
024 * <p>
025 * Instances are immutable: the minimum and maximum {@code byte[]} arrays are copied during
026 * construction and when accessed by {@link #getMin} and {@link #getMax}.
027 * </p>
028 */
029public class KeyRange implements SizeEstimating {
030
031    /**
032     * The {@link KeyRange} containing the full range (i.e., all keys).
033     */
034    public static final KeyRange FULL = new KeyRange(ByteUtil.EMPTY, null);
035
036    /**
037     * Sorts instances by {@linkplain KeyRange#getMin min value}, then {@linkplain KeyRange#getMax max value}.
038     */
039    public static final Comparator<KeyRange> SORT_BY_MIN = new Comparator<KeyRange>() {
040        @Override
041        public int compare(KeyRange keyRange1, KeyRange keyRange2) {
042            int diff = ByteUtil.compare(keyRange1.min, keyRange2.min);
043            if (diff != 0)
044                return diff;
045            diff = KeyRange.compare(keyRange1.max, keyRange2.max);
046            if (diff != 0)
047                return diff;
048            return 0;
049        }
050    };
051
052    /**
053     * Sorts instances by {@linkplain KeyRange#getMax max value}, then {@linkplain KeyRange#getMin min value}.
054     */
055    public static final Comparator<KeyRange> SORT_BY_MAX = new Comparator<KeyRange>() {
056        @Override
057        public int compare(KeyRange keyRange1, KeyRange keyRange2) {
058            int diff = KeyRange.compare(keyRange1.max, keyRange2.max);
059            if (diff != 0)
060                return diff;
061            diff = ByteUtil.compare(keyRange1.min, keyRange2.min);
062            if (diff != 0)
063                return diff;
064            return 0;
065        }
066    };
067
068    /**
069     * Lower bound (inclusive), or null for no minimum. Subclasses must <b>not</b> modify the array (to preserve immutability).
070     */
071    protected final byte[] min;
072
073    /**
074     * Upper bound (exclusive), or null for no maximum. Subclasses must <b>not</b> modify the array (to preserve immutability).
075     */
076    protected final byte[] max;
077
078// Constructors
079
080    /**
081     * Constructor.
082     *
083     * @param min minimum key (inclusive); must not be null
084     * @param max maximum key (exclusive), or null for no maximum
085     * @throws IllegalArgumentException if {@code min} is null
086     * @throws IllegalArgumentException if {@code min > max}
087     */
088    public KeyRange(byte[] min, byte[] max) {
089        if (min == null)
090            throw new IllegalArgumentException("null min");
091        if (KeyRange.compare(min, max) > 0)
092            throw new IllegalArgumentException("min = " + ByteUtil.toString(min) + " > max = " + ByteUtil.toString(max));
093        this.min = min.clone();
094        this.max = max != null ? max.clone() : null;
095    }
096
097    /**
098     * Construct key range containing a single key.
099     *
100     * @param key the key contained in the range
101     * @throws IllegalArgumentException if {@code key} is null
102     */
103    public KeyRange(byte[] key) {
104        if (key == null)
105            throw new IllegalArgumentException("null key");
106        this.min = key.clone();
107        this.max = ByteUtil.getNextKey(this.min);
108    }
109
110    /**
111     * Construct an instance containing all keys with the given prefix.
112     *
113     * @param prefix prefix of all keys in the range
114     * @return range of keys prefixed by {@code prefix}
115     * @throws IllegalArgumentException if {@code prefix} is null
116     */
117    public static KeyRange forPrefix(byte[] prefix) {
118        if (prefix == null)
119            throw new IllegalArgumentException("null prefix");
120        if (prefix.length == 0)
121            return KeyRange.FULL;
122        /*final*/ byte[] maxKey;
123        try {
124            maxKey = ByteUtil.getKeyAfterPrefix(prefix);
125        } catch (IllegalArgumentException e) {
126            maxKey = null;
127        }
128        return new KeyRange(prefix, maxKey);
129    }
130
131// Instance Methods
132
133    /**
134     * Get range minimum (inclusive).
135     *
136     * @return inclusivie minimum, never null
137     */
138    public byte[] getMin() {
139        return this.min.clone();
140    }
141
142    /**
143     * Get range maximum (exclusive), or null if there is no upper bound.
144     *
145     * @return exclusivie maximum, or null for none
146     */
147    public byte[] getMax() {
148        return this.max == null ? null : this.max.clone();
149    }
150
151    /**
152     * Determine if this key range overlaps the specified key range.
153     *
154     * @param range other instance
155     * @return true if this instance overlaps {@code range}
156     * @throws IllegalArgumentException if {@code range} is null
157     */
158    public boolean overlaps(KeyRange range) {
159        if (range == null)
160            throw new IllegalArgumentException("null range");
161        return KeyRange.compare(this.min, range.max) < 0 && KeyRange.compare(range.min, this.max) < 0;
162    }
163
164    /**
165     * Determine if this key range fully contains the specified key range.
166     *
167     * @param range other instance
168     * @return true if this instance contains {@code range}
169     * @throws IllegalArgumentException if {@code range} is null
170     */
171    public boolean contains(KeyRange range) {
172        if (range == null)
173            throw new IllegalArgumentException("null range");
174        return KeyRange.compare(this.min, range.min) <= 0 && KeyRange.compare(this.max, range.max) >= 0;
175    }
176
177    /**
178     * Determine if this key range contains the specified key.
179     *
180     * @param key key to test
181     * @return true if this range contains {@code key}
182     * @throws IllegalArgumentException if {@code key} is null
183     */
184    public boolean contains(byte[] key) {
185        return this.compareTo(key) == 0;
186    }
187
188    /**
189     * Determine whether this instance contains the full range covering all keys.
190     *
191     * @return true if this instance contains all keys
192     */
193    public boolean isFull() {
194        return this.min.length == 0 && this.max == null;
195    }
196
197    /**
198     * Determine whether this instance contains exactly one key.
199     *
200     * <p>
201     * If so, {@link #getMin} returns the key.
202     * </p>
203     *
204     * @return true if this instance contains exactly one key, otherwise false
205     */
206    public boolean isSingleKey() {
207        return this.max != null && this.max.length == this.min.length + 1 && this.max[this.min.length] == 0;
208    }
209
210    /**
211     * Determine whether this instance contains zero keys (implying {@link #getMin}{@code == }{@link #getMax}).
212     *
213     * @return true if this instance contains no keys
214     */
215    public boolean isEmpty() {
216        return this.max != null && ByteUtil.compare(this.min, this.max) == 0;
217    }
218
219    /**
220     * Create a new instance whose minimum and maximum keys are the same as this instance's
221     * but with the given byte sequence prepended.
222     *
223     * @param prefix key range prefix
224     * @return this range prefixed by {@code prefix}
225     * @throws IllegalArgumentException if {@code prefix} is null
226     */
227    public KeyRange prefixedBy(byte[] prefix) {
228        if (prefix == null)
229            throw new IllegalArgumentException("null prefix");
230        final byte[] prefixedMin = Bytes.concat(prefix, this.min);
231        /*final*/ byte[] prefixedMax;
232        if (this.max != null)
233            prefixedMax = Bytes.concat(prefix, this.max);
234        else {
235            try {
236                prefixedMax = ByteUtil.getKeyAfterPrefix(prefix);
237            } catch (IllegalArgumentException e) {
238                prefixedMax = null;
239            }
240        }
241        return new KeyRange(prefixedMin, prefixedMax);
242    }
243
244    /**
245     * Determine if this range is left of, contains, or is right of the given key.
246     *
247     * @param key key for comparison
248     * @return -1 if this range is left of {@code key},
249     *  0 if this range contains {@code key}, or 1 if this range is right of {@code key},
250     * @throws IllegalArgumentException if {@code key} is null
251     */
252    public int compareTo(byte[] key) {
253        if (key == null)
254            throw new IllegalArgumentException("null key");
255        if (KeyRange.compare(this.min, key) > 0)
256            return 1;
257        if (KeyRange.compare(this.max, key) <= 0)
258            return -1;
259        return 0;
260    }
261
262    /**
263     * Compare two {@code byte[]} keys using unsigned lexical ordering, while also accepting
264     * null values that represent "positive infinity".
265     *
266     * @param key1 first key
267     * @param key2 second key
268     * @return -1 if {@code key1 < key2}, 1 if {@code key1 > key2}, or zero if {@code key1 = key2}
269     */
270    public static int compare(byte[] key1, byte[] key2) {
271        if (key1 == null && key2 == null)
272            return 0;
273        if (key1 == null)
274            return 1;
275        if (key2 == null)
276            return -1;
277        return ByteUtil.compare(key1, key2);
278    }
279
280// SizeEstimating
281
282    @Override
283    public void addTo(SizeEstimator estimator) {
284        estimator
285          .addObjectOverhead()
286          .addField(this.min)
287          .addField(this.max);
288    }
289
290// Object
291
292    @Override
293    public boolean equals(Object obj) {
294        if (obj == this)
295            return true;
296        if (obj == null || obj.getClass() != this.getClass())
297            return false;
298        final KeyRange that = (KeyRange)obj;
299        return Arrays.equals(this.min, that.min) && (this.max == null ? that.max == null : Arrays.equals(this.max, that.max));
300    }
301
302    @Override
303    public int hashCode() {
304        return Arrays.hashCode(this.min) ^ (this.max != null ? Arrays.hashCode(this.max) : 0);
305    }
306
307    @Override
308    public String toString() {
309        return this.getClass().getSimpleName() + "[" + ByteUtil.toString(this.min) + "," + ByteUtil.toString(this.max) + "]";
310    }
311}
