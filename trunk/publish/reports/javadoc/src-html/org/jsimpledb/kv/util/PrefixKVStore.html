008package org.jsimpledb.kv.util;
009
010import com.google.common.base.Function;
011import com.google.common.base.Preconditions;
012import com.google.common.collect.Iterators;
013import com.google.common.primitives.Bytes;
014
015import java.util.Iterator;
016
017import org.jsimpledb.kv.KVPair;
018import org.jsimpledb.kv.KVStore;
019import org.jsimpledb.util.ByteUtil;
020
021/**
022 * A {@link org.jsimpledb.kv.KVStore} view of all keys having a common {@code byte[]} prefix
023 * in an outer, containing {@link org.jsimpledb.kv.KVStore}.
024 */
025public abstract class PrefixKVStore extends ForwardingKVStore {
026
027    private final byte[] keyPrefix;
028
029    /**
030     * Constructor.
031     *
032     * @param keyPrefix prefix for all keys
033     * @throws IllegalArgumentException if {@code keyPrefix} is null
034     */
035    public PrefixKVStore(byte[] keyPrefix) {
036        Preconditions.checkArgument(keyPrefix != null, "null keyPrefix");
037        this.keyPrefix = keyPrefix.clone();
038    }
039
040    /**
041     * Get the {@code byte[]} key prefix associated with this instance.
042     *
043     * @return (a copy of) this instance's key prefix
044     */
045    public final byte[] getKeyPrefix() {
046        return this.keyPrefix.clone();
047    }
048
049    /**
050     * Create a {@link PrefixKVStore} instance using the specified prefix and underlying {@link KVStore}.
051     *
052     * @param kvstore underyling key/value store
053     * @param keyPrefix prefix for all keys
054     * @return view of all keys in {@code kvstore} with prefix {@code keyPrefix}
055     * @throws IllegalArgumentException if either parameter is null
056     */
057    public static PrefixKVStore create(final KVStore kvstore, byte[] keyPrefix) {
058        Preconditions.checkArgument(kvstore != null, "null kvstore");
059        Preconditions.checkArgument(keyPrefix != null, "null keyPrefix");
060        return new PrefixKVStore(keyPrefix) {
061            @Override
062            protected KVStore delegate() {
063                return kvstore;
064            }
065        };
066    }
067
068// KVStore
069
070    @Override
071    public byte[] get(byte[] key) {
072        return this.delegate().get(this.addPrefix(key));
073    }
074
075    @Override
076    public KVPair getAtLeast(byte[] minKey) {
077        final KVPair pair = this.delegate().getAtLeast(this.addMinPrefix(minKey));
078        if (pair == null || !ByteUtil.isPrefixOf(this.keyPrefix, pair.getKey()))
079            return null;
080        return new KVPair(this.removePrefix(pair.getKey()), pair.getValue());
081    }
082
083    @Override
084    public KVPair getAtMost(byte[] maxKey) {
085        final KVPair pair = this.delegate().getAtMost(this.addMaxPrefix(maxKey));
086        if (pair == null || !ByteUtil.isPrefixOf(this.keyPrefix, pair.getKey()))
087            return null;
088        return new KVPair(this.removePrefix(pair.getKey()), pair.getValue());
089    }
090
091    @Override
092    public Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, boolean reverse) {
093        final Iterator<KVPair> i = this.delegate().getRange(this.addMinPrefix(minKey), this.addMaxPrefix(maxKey), reverse);
094        return Iterators.transform(i, new Function<KVPair, KVPair>() {
095            @Override
096            public KVPair apply(KVPair pair) {
097                return new KVPair(PrefixKVStore.this.removePrefix(pair.getKey()), pair.getValue());
098            }
099        });
100    }
101
102    @Override
103    public void put(byte[] key, byte[] value) {
104        this.delegate().put(this.addPrefix(key), value);
105    }
106
107    @Override
108    public void remove(byte[] key) {
109        this.delegate().remove(this.addPrefix(key));
110    }
111
112    @Override
113    public void removeRange(byte[] minKey, byte[] maxKey) {
114        this.delegate().removeRange(this.addMinPrefix(minKey), this.addMaxPrefix(maxKey));
115    }
116
117    @Override
118    public void adjustCounter(byte[] key, long amount) {
119        this.delegate().adjustCounter(this.addPrefix(key), amount);
120    }
121
122// Key (un)prefixing
123
124    private byte[] addPrefix(byte[] key) {
125        if (key == null)
126            return null;
127        return Bytes.concat(this.keyPrefix, key);
128    }
129
130    private byte[] addMinPrefix(byte[] minKey) {
131        if (minKey == null)
132            return this.keyPrefix.clone();
133        return this.addPrefix(minKey);
134    }
135
136    private byte[] addMaxPrefix(byte[] maxKey) {
137        if (maxKey == null)
138            return this.keyPrefix.length > 0 ? ByteUtil.getKeyAfterPrefix(this.keyPrefix) : null;
139        return this.addPrefix(maxKey);
140    }
141
142    private byte[] removePrefix(byte[] key) {
143        if (!ByteUtil.isPrefixOf(this.keyPrefix, key)) {
144            throw new IllegalArgumentException("read key " + ByteUtil.toString(key) + " not having "
145              + ByteUtil.toString(this.keyPrefix) + " as a prefix");
146        }
147        final byte[] suffix = new byte[key.length - this.keyPrefix.length];
148        System.arraycopy(key, this.keyPrefix.length, suffix, 0, suffix.length);
149        return suffix;
150    }
151}
