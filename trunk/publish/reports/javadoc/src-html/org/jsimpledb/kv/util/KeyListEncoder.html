008package org.jsimpledb.kv.util;
009
010import com.google.common.base.Preconditions;
011
012import java.io.EOFException;
013import java.io.IOException;
014import java.io.InputStream;
015import java.io.OutputStream;
016
017import org.jsimpledb.util.LongEncoder;
018import org.jsimpledb.util.UnsignedIntEncoder;
019
020/**
021 * Serializes a sequence of {@code byte[]} arrays, compressing common prefixes.
022 *
023 * <p>
024 * Keys are encoded in one of two forms:
025 *  <ul>
026 *  <li>{@code total-length bytes...}
027 *  <li>{@code -prefix-length suffix-length suffix-bytes ...}
028 *  </ul>
029 * The first length ({@code total-length} or negative {@code prefix-length}) is encoded using {@link LongEncoder}.
030 * The {@code suffix-length}, if present, is encoded using {@link UnsignedIntEncoder}.
031 */
032public final class KeyListEncoder {
033
034    private KeyListEncoder() {
035    }
036
037    /**
038     * Write the next key, compressing its common prefix with the previous key (if any).
039     *
040     * @param out output stream
041     * @param key key to write
042     * @param prev previous key, or null for none
043     * @throws IOException if an I/O error occurs
044     * @throws IllegalArgumentException if {@code out} or {@code key} is null
045     */
046    public static void write(OutputStream out, byte[] key, byte[] prev) throws IOException {
047        Preconditions.checkArgument(out != null, "null out");
048        Preconditions.checkArgument(key != null, "null key");
049        int prefixLength = 0;
050        if (prev != null) {
051            while (prefixLength < key.length && prefixLength < prev.length && key[prefixLength] == prev[prefixLength])
052                prefixLength++;
053        }
054        if (prefixLength > 1) {
055            final int suffixLength = key.length - prefixLength;
056            LongEncoder.write(out, ~(prefixLength - 2));
057            UnsignedIntEncoder.write(out, suffixLength);
058            out.write(key, prefixLength, suffixLength);
059        } else {
060            LongEncoder.write(out, key.length);
061            out.write(key);
062        }
063    }
064
065    /**
066     * Calculate the number of bytes that would be required to write the next key via {@link #write write()}.
067     *
068     * @param key key to write
069     * @param prev previous key, or null for none
070     * @return number of bytes to be written by {@link #write write(out, key, prev)}
071     * @throws IllegalArgumentException if {@code key} is null
072     */
073    public static int writeLength(byte[] key, byte[] prev) {
074        Preconditions.checkArgument(key != null, "null key");
075        int prefixLength = 0;
076        if (prev != null) {
077            while (prefixLength < key.length && prefixLength < prev.length && key[prefixLength] == prev[prefixLength])
078                prefixLength++;
079        }
080        if (prefixLength > 1) {
081            final int suffixLength = key.length - prefixLength;
082            return LongEncoder.encodeLength(~(prefixLength - 2)) + UnsignedIntEncoder.encodeLength(suffixLength) + suffixLength;
083        } else
084            return LongEncoder.encodeLength(key.length) + key.length;
085    }
086
087    /**
088     * Read the next key.
089     *
090     * @param input input stream
091     * @param prev previous key, or null for none
092     * @return next key
093     * @throws IOException if an I/O error occurs
094     * @throws EOFException if an unexpected EOF is encountered
095     * @throws IllegalArgumentException if {@code input} is null
096     * @throws IllegalArgumentException if {@code input} contains invalid data
097     */
098    public static byte[] read(InputStream input, byte[] prev) throws IOException {
099        Preconditions.checkArgument(input != null, "null input");
100
101        // Get encoded length of prefix
102        int keyLength = KeyListEncoder.readSignedInt(input);
103        final byte[] key;
104        int prefixLength;
105
106        // Decode prefix length and read prefix, if any
107        if (keyLength < 0) {
108            if (prev == null)
109                throw new IllegalArgumentException("null `prev' given but next key has " + -keyLength + " byte shared prefix");
110            prefixLength = ~keyLength + 2;
111            final int suffixLength = UnsignedIntEncoder.read(input);
112            keyLength = prefixLength + suffixLength;
113            if (keyLength < 0)
114                throw new IllegalArgumentException("invalid prefix length " + prefixLength + " plus suffix length " + suffixLength);
115            key = new byte[keyLength];
116            System.arraycopy(prev, 0, key, 0, prefixLength);
117        } else {
118            key = new byte[keyLength];
119            prefixLength = 0;
120        }
121
122        // Read suffix
123        while (prefixLength < key.length) {
124            final int num = input.read(key, prefixLength, key.length - prefixLength);
125            if (num == -1)
126                throw new EOFException();
127            prefixLength += num;
128        }
129
130        // Done
131        return key;
132    }
133
134    private static int readSignedInt(InputStream input) throws IOException {
135        final long longValue = LongEncoder.read(input);
136        final int intValue = (int)longValue;
137        Preconditions.checkArgument(intValue == (int)longValue, "read out-of-range encoded int value %s", longValue);
138        return intValue;
139    }
140}
