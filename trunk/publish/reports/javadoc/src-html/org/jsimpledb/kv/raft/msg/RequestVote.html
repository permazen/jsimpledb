008package org.jsimpledb.kv.raft.msg;
009
010import com.google.common.base.Preconditions;
011
012import java.nio.ByteBuffer;
013
014import org.jsimpledb.util.LongEncoder;
015
016/**
017 * Send from candidates to other nodes to request a vote during an election.
018 */
019public class RequestVote extends Message {
020
021    private final long lastLogTerm;
022    private final long lastLogIndex;
023
024// Constructors
025
026    /**
027     * Constructor.
028     *
029     * @param clusterId cluster ID
030     * @param senderId identity of sender
031     * @param recipientId identity of recipient
032     * @param term sender's current term
033     * @param lastLogTerm term of the sender's last log entry
034     * @param lastLogIndex index of the sender's last log entry
035     */
036    public RequestVote(int clusterId, String senderId, String recipientId, long term, long lastLogTerm, long lastLogIndex) {
037        super(Message.REQUEST_VOTE_TYPE, clusterId, senderId, recipientId, term);
038        this.lastLogTerm = lastLogTerm;
039        this.lastLogIndex = lastLogIndex;
040        this.checkArguments();
041    }
042
043    RequestVote(ByteBuffer buf) {
044        super(Message.REQUEST_VOTE_TYPE, buf);
045        this.lastLogTerm = LongEncoder.read(buf);
046        this.lastLogIndex = LongEncoder.read(buf);
047        this.checkArguments();
048    }
049
050    @Override
051    void checkArguments() {
052        super.checkArguments();
053        Preconditions.checkArgument(this.lastLogTerm > 0);
054        Preconditions.checkArgument(this.lastLogIndex > 0);
055    }
056
057// Properties
058
059    public long getLastLogTerm() {
060        return this.lastLogTerm;
061    }
062
063    public long getLastLogIndex() {
064        return this.lastLogIndex;
065    }
066
067// Message
068
069    @Override
070    public void visit(MessageSwitch handler) {
071        handler.caseRequestVote(this);
072    }
073
074    @Override
075    public void writeTo(ByteBuffer dest) {
076        super.writeTo(dest);
077        LongEncoder.write(dest, this.lastLogTerm);
078        LongEncoder.write(dest, this.lastLogIndex);
079    }
080
081    @Override
082    protected int calculateSize() {
083        return super.calculateSize()
084          + LongEncoder.encodeLength(this.lastLogTerm)
085          + LongEncoder.encodeLength(this.lastLogIndex);
086    }
087
088// Object
089
090    @Override
091    public String toString() {
092        return this.getClass().getSimpleName()
093          + "[\"" + this.getSenderId() + "\"->\"" + this.getRecipientId() + "\""
094          + ",clusterId=" + String.format("%08x", this.getClusterId())
095          + ",term=" + this.getTerm()
096          + ",lastLogTerm=" + this.lastLogTerm
097          + ",lastLogIndex=" + this.lastLogIndex
098          + "]";
099    }
100}
