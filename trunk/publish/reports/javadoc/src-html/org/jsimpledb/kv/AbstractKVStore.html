008package org.jsimpledb.kv;
009
010import java.util.Arrays;
011import java.util.Iterator;
012
013import org.jsimpledb.util.ByteReader;
014import org.jsimpledb.util.ByteUtil;
015import org.jsimpledb.util.ByteWriter;
016
017/**
018 * Support superclass for {@link KVStore} implementations.
019 *
020 * <p>
021 * This class provides a partial implementation via the following methods:
022 * <ul>
023 *  <li>A {@link #get get()} implementation based on {@link #getAtLeast getAtLeast()}</li>
024 *  <li>{@link #getAtLeast getAtLeast()} and {@link #getAtMost getAtMost()} implementations based on
025 *      {@link #getRange getRange()}.</li>
026 *  <li>A {@link #remove remove()} implementation that delegates to {@link #removeRange removeRange()}.</li>
027 *  <li>A {@link #removeRange removeRange()} implementation that delegates to {@link #getRange getRange()},
028 *      iterating through the range of keys and removing them one-by-one via {@link Iterator#remove}.</li>
029 *  <li>{@link #encodeCounter encodeCounter()}, {@link #decodeCounter encodeCounter()}, and
030 *      {@link #adjustCounter adjustCounter()} implementations using normal reads and writes
031 *      of values in big-endian encoding (does not provide any lock-free behavior).</li>
032 *  <li>A {@link #put put()} implementation throwing {@link UnsupportedOperationException}</li>
033 * </ul>
034 *
035 * <p>
036 * Therefore, a read-only {@link KVStore} implementation is possible simply by implementing {@link #getRange}.
037 * </p>
038 */
039public abstract class AbstractKVStore implements KVStore {
040
041    protected AbstractKVStore() {
042    }
043
044    @Override
045    public byte[] get(byte[] key) {
046        final KVPair pair = this.getAtLeast(key);
047        return pair != null && Arrays.equals(pair.getKey(), key) ? pair.getValue() : null;
048    }
049
050    @Override
051    public KVPair getAtLeast(byte[] minKey) {
052        final Iterator<KVPair> i = this.getRange(minKey, null, false);
053        try {
054            return i.hasNext() ? i.next() : null;
055        } finally {
056            this.closeIfPossible(i);
057        }
058    }
059
060    @Override
061    public KVPair getAtMost(byte[] maxKey) {
062        final Iterator<KVPair> i = this.getRange(null, maxKey, true);
063        try {
064            return i.hasNext() ? i.next() : null;
065        } finally {
066            this.closeIfPossible(i);
067        }
068    }
069
070    @Override
071    public void put(byte[] key, byte[] value) {
072        throw new UnsupportedOperationException();
073    }
074
075    @Override
076    public void remove(byte[] key) {
077        this.removeRange(key, ByteUtil.getNextKey(key));
078    }
079
080    @Override
081    public void removeRange(byte[] minKey, byte[] maxKey) {
082        final Iterator<KVPair> i = this.getRange(minKey, maxKey, false);
083        try {
084            while (i.hasNext()) {
085                i.next();
086                i.remove();
087            }
088        } finally {
089            this.closeIfPossible(i);
090        }
091    }
092
093    @Override
094    public byte[] encodeCounter(long value) {
095        final ByteWriter writer = new ByteWriter(8);
096        ByteUtil.writeLong(writer, value);
097        return writer.getBytes();
098    }
099
100    @Override
101    public long decodeCounter(byte[] value) {
102        if (value.length != 8)
103            throw new IllegalArgumentException("invalid encoded counter value: length = " + value.length + " != 8");
104        return ByteUtil.readLong(new ByteReader(value));
105    }
106
107    @Override
108    public void adjustCounter(byte[] key, long amount) {
109        if (key == null)
110            throw new NullPointerException("null key");
111        final byte[] previous = this.get(key);
112        if (previous == null)
113            return;
114        this.put(key, this.encodeCounter(this.decodeCounter(previous) + amount));
115    }
116
117    private void closeIfPossible(Iterator<KVPair> i) {
118        if (i instanceof AutoCloseable) {
119            try {
120                ((AutoCloseable)i).close();
121            } catch (Exception e) {
122                // ignore
123            }
124        }
125    }
126}
