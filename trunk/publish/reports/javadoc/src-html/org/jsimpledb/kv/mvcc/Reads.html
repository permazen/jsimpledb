008package org.jsimpledb.kv.mvcc;
009
010import com.google.common.base.Converter;
011import com.google.common.base.Preconditions;
012
013import java.io.IOException;
014import java.io.InputStream;
015import java.io.OutputStream;
016import java.util.ArrayList;
017import java.util.Map;
018
019import org.jsimpledb.kv.KeyRange;
020import org.jsimpledb.kv.KeyRanges;
021import org.jsimpledb.util.ByteUtil;
022import org.jsimpledb.util.SizeEstimating;
023import org.jsimpledb.util.SizeEstimator;
024
025/**
026 * Holds a set of reads from a {@link org.jsimpledb.kv.KVStore}.
027 *
028 * <p>
029 * Only the (ranges of) keys read are retained, not the values.
030 *
031 * <p>
032 * Instances are not thread safe.
033 * </p>
034 */
035public class Reads implements SizeEstimating {
036
037    private KeyRanges reads;
038
039// Constructors
040
041    /**
042     * Constructs an empty instance.
043     */
044    public Reads() {
045        this(KeyRanges.EMPTY);
046    }
047
048    /**
049     * Constructs an instance initialized with the given read ranges
050     *
051     * @param reads read ranges
052     * @throws IllegalArgumentException if {@code reads} is null
053     */
054    public Reads(KeyRanges reads) {
055        this.setReads(reads);
056    }
057
058// Public methods
059
060    /**
061     * Get the ranges of keys read.
062     *
063     * @return ranges of keys read
064     */
065    public KeyRanges getReads() {
066        return this.reads;
067    }
068
069    /**
070     * Set the ranges of keys read.
071     *
072     * @param reads ranges of keys read
073     * @throws IllegalArgumentException if {@code reads} is null
074     */
075    public void setReads(KeyRanges reads) {
076        Preconditions.checkArgument(reads != null, "null reads");
077        this.reads = reads;
078    }
079
080// MVCC
081
082    /**
083     * Determine whether any of the given mutations conflict with any of the keys read by this instance.
084     *
085     * <p>
086     * If this method returns false, then if two transactions T1 and T2 are based on the same underlying
087     * {@link org.jsimpledb.kv.KVStore} snapshot, and T1 writes {@code mutations} and T2 reads according to this instance,
088     * then T2 can be ordered after T1 while still preserving linearizable semantics. That is, the given {@code mutations}
089     * are invisible to this instance.
090     *
091     * <p>
092     * This method guarantees that it will access the given {@code mutations} in this order: removes, puts, adjusts.
093     *
094     * @param mutations mutations to check for conflicts
095     * @return true if the {@code mutations} are invisible to this instance, false if there is a read/write conflict
096     * @throws IllegalArgumentException if {@code mutations} is null
097     */
098    public boolean isConflict(Mutations mutations) {
099        Preconditions.checkArgument(mutations != null, "null mutations");
100
101        // Check removes
102        final ArrayList<KeyRange> removes = new ArrayList<KeyRange>();
103        for (KeyRange remove : mutations.getRemoveRanges())
104            removes.add(remove);
105        if (!this.reads.intersection(new KeyRanges(removes)).isEmpty())
106            return true;                        // read/remove conflict
107
108        // Check puts
109        for (Map.Entry<byte[], byte[]> entry : mutations.getPutPairs()) {
110            if (this.reads.contains(entry.getKey()))
111                return true;                    // read/write conflict
112        }
113
114        // Check adjusts
115        for (Map.Entry<byte[], Long> entry : mutations.getAdjustPairs()) {
116            if (this.reads.contains(entry.getKey()))
117                return true;                    // read/adjust conflict
118        }
119
120        // No conflicts
121        return false;
122    }
123
124// Serialization
125
126    /**
127     * Serialize this instance.
128     *
129     * @param out output
130     * @throws IOException if an error occurs
131     */
132    public void serialize(OutputStream out) throws IOException {
133        this.reads.serialize(out);
134    }
135
136    /**
137     * Calculate the number of bytes required to serialize this instance via {@link #serialize serialize()}.
138     *
139     * @return number of serialized bytes
140     */
141    public long serializedLength() {
142        return this.reads.serializedLength();
143    }
144
145    /**
146     * Deserialize an instance created by {@link #serialize serialize()}.
147     *
148     * @param input input stream containing data from {@link #serialize serialize()}
149     * @return deserialized instance
150     * @throws IOException if an I/O error occurs
151     * @throws java.io.EOFException if the input ends unexpectedly
152     * @throws IllegalArgumentException if malformed input is detected
153     * @throws IllegalArgumentException if {@code input} is null
154     */
155    public static Reads deserialize(InputStream input) throws IOException {
156        Preconditions.checkArgument(input != null, "null input");
157        return new Reads(KeyRanges.deserialize(input));
158    }
159
160// SizeEstimating
161
162    @Override
163    public void addTo(SizeEstimator estimator) {
164        estimator
165          .addObjectOverhead()
166          .addField(this.reads);
167    }
168
169// Object
170
171    @Override
172    public String toString() {
173        final Converter<String, byte[]> byteConverter = ByteUtil.STRING_CONVERTER.reverse();
174        return this.getClass().getSimpleName()
175          + "[reads=" + reads
176          + "]";
177    }
178}
