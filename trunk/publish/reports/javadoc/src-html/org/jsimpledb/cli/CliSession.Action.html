041/**
042 * Represents one CLI console session.
043 */
044public class CliSession extends ParseSession {
045
046    private final PrintWriter writer;
047    private final TreeMap<String, AbstractCommand> commands = new TreeMap<>();
048
049    private boolean done;
050    private boolean verbose;
051    private int lineLimit = 16;
052
053// Constructors
054
055    /**
056     * Constructor for core level access only.
057     *
058     * @param db core database
059     * @param writer console output
060     * @throws IllegalArgumentException if either parameter is null
061     */
062    public CliSession(Database db, PrintWriter writer) {
063        super(db);
064        if (writer == null)
065            throw new IllegalArgumentException("null writer");
066        this.writer = writer;
067    }
068
069    /**
070     * Constructor for {@link JSimpleDB} level access.
071     *
072     * @param jdb database
073     * @param writer console output
074     * @throws IllegalArgumentException if either parameter is null
075     */
076    public CliSession(JSimpleDB jdb, PrintWriter writer) {
077        super(jdb);
078        if (writer == null)
079            throw new IllegalArgumentException("null writer");
080        this.writer = writer;
081    }

083// Accessors
084

088     * @return output writer

094    /**

115    }

118        return this.done;
119    }

121        this.done = done;

124// Command registration
125
126    /**
127     * Register the standard CLI built-in functions such as {@code print()}, {@code dump()}, etc.
128     */
129    @Override
130    public void registerStandardFunctions() {
131        super.registerStandardFunctions();
132
133        // We don't use AnnotatedClassScanner here to avoid having a dependency on the spring classes
134        this.registerFunction(DumpFunction.class);
135        this.registerFunction(PrintFunction.class);
136    }
137
138    /**
139     * Register the standard CLI built-in commands.
140     */
141    public void registerStandardCommands() {
142
143        // We don't use AnnotatedClassScanner here to avoid having a dependency on the spring classes
144        this.registerCommand(CompareSchemasCommand.class);
145        this.registerCommand(DeleteSchemaVersionCommand.class);
146        this.registerCommand(EvalCommand.class);
147        this.registerCommand(HelpCommand.class);
148        this.registerCommand(ImportCommand.class);
149        this.registerCommand(InfoCommand.class);
150        this.registerCommand(LoadCommand.class);
151        this.registerCommand(QuitCommand.class);
152        this.registerCommand(RaftAddCommand.class);
153        this.registerCommand(RaftRemoveCommand.class);
154        this.registerCommand(SaveCommand.class);
155        this.registerCommand(SetAllowNewSchemaCommand.class);
156        this.registerCommand(SetSchemaVersionCommand.class);
157        this.registerCommand(SetValidationModeCommand.class);
158        this.registerCommand(ShowAllSchemasCommand.class);
159        this.registerCommand(ShowSchemaCommand.class);
160    }
161
162    /**
163     * Create an instance of the specified class and register it as a {@link AbstractCommand}.
164     * The class must have a public constructor taking either a single {@link CliSession} parameter
165     * or no parameters; they will be tried in that order.
166     *
167     * @param cl command class
168     * @throws IllegalArgumentException if {@code cl} has no suitable constructor
169     * @throws IllegalArgumentException if {@code cl} instantiation fails
170     * @throws IllegalArgumentException if {@code cl} does not subclass {@link AbstractCommand}
171     */
172    public void registerCommand(Class<?> cl) {
173        if (!AbstractCommand.class.isAssignableFrom(cl))
174            throw new IllegalArgumentException(cl + " does not subclass " + AbstractCommand.class.getName());
175        final AbstractCommand command = this.instantiate(cl.asSubclass(AbstractCommand.class));
176        this.commands.put(command.getName(), command);
177    }
178
179    private <T> T instantiate(Class<T> cl) {
180        Throwable failure;
181        try {
182            return cl.getConstructor(CliSession.class).newInstance(this);
183        } catch (NoSuchMethodException e) {
184            try {
185                return cl.getConstructor().newInstance();
186            } catch (NoSuchMethodException e2) {
187                throw new IllegalArgumentException("no suitable constructor found in class " + cl.getName());
188            } catch (Exception e2) {
189                failure = e2;
190            }
191        } catch (Exception e) {
192            failure = e;
193        }
194        if (failure instanceof InvocationTargetException)
195            failure = failure.getCause();
196        throw new IllegalArgumentException("unable to instantiate class " + cl.getName() + ": " + failure, failure);
197    }
198
199// Errors
200
201    @Override
202    protected void reportException(Exception e) {
203        final String message = e.getLocalizedMessage();
204        if (e instanceof ParseException && message != null)
205            this.writer.println("Error: " + message);
206        else
207            this.writer.println("Error: " + e.getClass().getSimpleName() + (message != null ? ": " + message : ""));
208        if (this.verbose || this.showStackTrace(e))
209            e.printStackTrace(this.writer);
210    }
211
212    protected boolean showStackTrace(Exception e) {
213        return e instanceof NullPointerException || (e instanceof ParseException && e.getLocalizedMessage() == null);
214    }
215
216// Action
217
218    /**
219     * Perform the given action. This is a convenience method, equivalent to: {@code perform(null, action)}
220     *
221     * @param action action to perform
222     * @return true if {@code action} completed successfully, false if the transaction could not be created
223     *  or {@code action} threw an exception
224     * @throws IllegalArgumentException if {@code action} is null
225     */
226    public boolean perform(Action action) {
227        return this.perform(null, action);
228    }
229
230    /**
231     * Perform the given action within the given existing transaction, if any, otherwise within a new transaction.
232     * If {@code tx} is not null, it will used and left open when this method returns. Otherwise,
233     * if there is already an open transaction associated with this instance, it will be used;
234     * otherwise, a new transaction is created for the duration of {@code action} and then committed.
235     *
236     * <p>
237     * If {@code tx} is not null and there is already an open transaction associated with this instance and they
238     * are not the same transaction, an {@link IllegalStateException} is thrown.
239     * </p>
240     *
241     * @param tx transaction in which to perform the action, or null to create a new one (if necessary)
242     * @param action action to perform
243     * @return true if {@code action} completed successfully, false if the transaction could not be created
244     *  or {@code action} threw an exception
245     * @throws IllegalStateException if {@code tx} conflict with the already an open transaction associated with this instance
246     * @throws IllegalArgumentException if {@code action} is null
247     */
248    public boolean perform(Transaction tx, final Action action) {
249        return this.perform(tx, new Session.Action() {
250            @Override
251            public void run(Session session) throws Exception {
252                action.run((CliSession)session);
253            }
254        });
255    }
256
257    /**
258     * Callback interface used by {@link CliSession#perform CliSession.perform()}.
259     */
260    public interface Action {
261
262        /**
263         * Perform some action using the given {@link CliSession} while a transaction is open.
264         *
265         * @param session session with open transaction
266         * @throws Exception if an error occurs
267         */
268        void run(CliSession session) throws Exception;
269    }
270}
